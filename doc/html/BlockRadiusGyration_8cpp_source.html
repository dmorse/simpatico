178  * Evaluate end-to-end vectors of all chains, add to ensemble.
179  */
180  void BlockRadiusGyration::sample(long iStep)
181  {
182  if (isAtInterval(iStep)) {
183 
184  Molecule* moleculePtr;
185  Vector r1, r2, dR;
186  int i, j, k, l, m, typeId, nMolecule;
187 
188  k = 0;
189  for (i = 0; i < nAtomType_; ++i) {
190  dRSq_[i] = 0.0;
191  iTypeNAtom_[i] = 0;
192  for (j = i+1; j < nAtomType_; ++j) {
193  ++k;
194  dRSqPair_[k-1] = 0.0;
195  }
196  }
197 
198  nMolecule = system().nMolecule(speciesId_);
199  moleculePtr = &system().molecule(speciesId_,0);
200  for (j = 0 ; j < nAtom_; j++) {
201  typeId = moleculePtr->atom(j).typeId();
202  iTypeNAtom_[typeId] += 1;
203  }
204 
205  for (i = 0; i < system().nMolecule(speciesId_); i++) {
206  moleculePtr = &system().molecule(speciesId_, i);
207 
208  // Initializing COM positions for different types to zero
209  for (j = 0; j < nAtomType_; ++j) {
210  rCom_[j].zero();
211  }
212 
213  // Construct map of molecule with no periodic boundary conditions
214  positions_[0] = moleculePtr->atom(0).position();
215  typeId = moleculePtr->atom(0).typeId();
216  rCom_[typeId] += positions_[0];
217  for (j = 1 ; j < nAtom_; j++) {
218  typeId = moleculePtr->atom(j).typeId();
219  r1 = moleculePtr->atom(j-1).position();
220  r2 = moleculePtr->atom(j).position();
221  system().boundary().distanceSq(r1, r2, dR);
222  positions_[j] = positions_[j-1];
223  positions_[j] += dR;
224  rCom_[typeId] += positions_[j];
225  }
226  for (l = 0; l < nAtomType_; ++l) {
227  rCom_[l] /= double(iTypeNAtom_[l]);
228  }
229  k = 0;
230  for (l = 0; l < nAtomType_; ++l) {
231  for (m = l+1; m < nAtomType_; ++m) {
232  ++k;
233  dR.subtract(rCom_[l], rCom_[m]);
234  dRSqPair_[k-1] += dR.square();
235  }
236  }
237  for (j = 0 ; j < nAtom_; j++) {
238  typeId = moleculePtr->atom(j).typeId();
239  dR.subtract(positions_[j], rCom_[typeId]);
240  dRSq_[typeId] += dR.square();
241  }
242 
243  }
244  k = 0;
245  for (i = 0; i < nAtomType_; ++i) {
246  dRSq_[i] /= double(nMolecule);
247  dRSq_[i] /= double(iTypeNAtom_[i]);
248  accumulators_[i].sample(dRSq_[i]);
249  outputFile_ << Dbl(dRSq_[i]) << " ";
250  for (j = i+1; j < nAtomType_; ++j) {
251  ++k;
252  dRSqPair_[k-1] /= double(nMolecule);
253  accumulators_[nAtomType_+k-1].sample(dRSqPair_[k-1]);
254  outputFile_ << Dbl(dRSqPair_[k-1]) << " ";
255  }
256  }
257  outputFile_ << std::endl;
258  } // if isAtInterval
259 
260  }
