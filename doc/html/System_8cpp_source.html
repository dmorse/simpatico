618  /*
619  * Load configuration from an archive.
620  */
621  void System::loadConfig(Serializable::IArchive &ar)
622  {
623  ar >> boundary();
624 
625  Species* speciesPtr;
626  int nSpecies = simulation().nSpecies();
627  Molecule* molPtr;
628  Molecule::AtomIterator atomIter;
629  int iSpeciesIn, nMoleculeIn;
630 
631  for (int iSpecies = 0; iSpecies < nSpecies; ++iSpecies) {
632  speciesPtr = &simulation().species(iSpecies);
633  ar >> iSpeciesIn;
634  if (iSpeciesIn != iSpecies) {
635  UTIL_THROW("Error: iSpeciesIn != iSpecies");
636  }
637  ar >> nMoleculeIn;
638  for (int iMol = 0; iMol < nMoleculeIn; ++iMol) {
639  molPtr = &(speciesPtr->reservoir().pop());
640  addMolecule(*molPtr);
641  if (molPtr != &molecule(iSpecies, iMol)) {
642  UTIL_THROW("Molecule index error");
643  }
644  molPtr->begin(atomIter);
645  for ( ; atomIter.notEnd(); ++atomIter) {
646  ar >> atomIter->position();
647  ar >> atomIter->velocity();
648  #ifdef MCMD_SHIFT
649  ar >> atomIter->shift();
650  boundary().shift(atomIter->position(), atom.shift());
651  #else
652  boundary().shift(atomIter->position());
653  #endif
654  }
655  }
656  }
657  }

659  /*
660  * Save internal state to an archive.
661  */
662  void System::saveConfig(Serializable::OArchive& ar)
663  {
664  ar << boundary();
665 
666  int nSpecies = simulation().nSpecies();
667  System::MoleculeIterator molIter;
668  Molecule::AtomIterator atomIter;
669  int nMoleculeOut;
670 
671  for (int iSpecies = 0; iSpecies < nSpecies; ++iSpecies) {
672  ar << iSpecies;
673  nMoleculeOut = nMolecule(iSpecies);
674  ar << nMoleculeOut;
675  begin(iSpecies, molIter);
676  for ( ; molIter.notEnd(); ++molIter) {
677  molIter->begin(atomIter);
678  for ( ; atomIter.notEnd(); ++atomIter) {
679  #ifdef MCMD_SHIFT
680  boundary().shift(atomIter->position(), atomIter->shift());
681  #else
682  boundary().shift(atomIter->position());
683  #endif
684  ar << atomIter->position();
685  ar << atomIter->velocity();
686  #ifdef MCMD_SHIFT
687  ar << atomIter->shift();
688  #endif
689  }
690  }
691  }
