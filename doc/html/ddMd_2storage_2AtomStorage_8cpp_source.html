116  UTIL_THROW("AtomStorage can only be initialized once");
117  }
118 
119  atoms_.allocate(atomCapacity_);

140  /*
141  * Zero forces on all local atoms and optionally on ghosts.
142  */
143  void AtomStorage::zeroForces(bool zeroGhosts)
144  {
145  int factor = 2;
146 
147  // Zero forces for all local atoms
148  if (nAtom() > atomCapacity_/factor) {
149  atoms_.zeroForces();
150  // Optimization to allow sequential access
151  } else {
152  AtomIterator atomIter;
153  begin(atomIter);
154  for( ; atomIter.notEnd(); ++atomIter){
155  atomIter->force().zero();
156  }
157  }
158 
159  // If using reverse communication, zero ghost atoms
160  if (zeroGhosts && nGhost()) {
161  if (nGhost() > ghostCapacity_/factor) {
162  ghosts_.zeroForces();
163  // Optimization to allow sequential access
164  } else {
165  GhostIterator ghostIter;
166  begin(ghostIter);
167  for( ; ghostIter.notEnd(); ++ghostIter){
168  ghostIter->force().zero();
169  }
170  }
171  }

393  /*
394  * Return max. sq. displacement of local atoms on this node since snapshot.
395  */
396  double AtomStorage::maxSqDisplacement()
397  {
398  if (!isCartesian()) {
399  UTIL_THROW("Error: Coordinates not Cartesian in maxSqDisplacement");
400  }
401  if (!locked_) {
402  UTIL_THROW("Error: AtomStorage not locked in maxSqDisplacement");
403  }
404  Vector dr;
405  double norm;
406  double max = 0.0;
407  AtomIterator iter;
408  int i = 0;
409  for (begin(iter); iter.notEnd(); ++iter) {
410  dr.subtract(iter->position(), snapshot_[i]);
411  norm = dr.square();
412  if (norm > max) {
413  max = norm;
414  }
415  ++i;
416  }
417  return max;
418  }

580  /*
581  * Check validity of this AtomStorage.
582  *
583  * Returns true if all is ok, or throws an Exception.
584  */
585  bool AtomStorage::isValid() const
586  {
587  if (nAtom() + atomReservoir_.size() != atomCapacity_) {
588  UTIL_THROW("nAtom + reservoir size != local capacity");
589  }
590  if (nGhost() + ghostReservoir_.size() != ghostCapacity_) {
591  UTIL_THROW("nGhost + reservoir size != ghost capacity");
592  }
593  if (nGhost() != map_.nGhost()) {
594  UTIL_THROW("Inconsistent values of nGhost");
595  }
596 
597  // Test validity of AtomMap.
598  map_.isValid();
599 
600  // Iterate over, count and find local atoms on this processor.
601  ConstAtomIterator localIter;
602  Atom* ptr;
603  int j = 0;
604  for (begin(localIter); localIter.notEnd(); ++localIter) {
605  ++j;
606  if (localIter->isGhost()) {
607  UTIL_THROW("Atom in atomSet is marked isGhost");
608  }
609  ptr = map_.find(localIter->id());
610  if (ptr == 0) {
611  UTIL_THROW("Unable to find local atom returned by iterator");
612  }
613  if (ptr != localIter.get()) {
614  UTIL_THROW("Inconsistent find(localIter->id()");
615  }
616  }
617  if (j != nAtom()) {
618  UTIL_THROW("Number counted by localIterator != nAtom()");
619  }
620 
621  // Iterate over, count and find ghost atoms
622  ConstGhostIterator ghostIter;
623  j = 0;
624  for (begin(ghostIter); ghostIter.notEnd(); ++ghostIter) {
625  ++j;
626  if (!ghostIter->isGhost()) {
627  UTIL_THROW("Atom in ghostSet is not marked isGhost");
628  }
629  ptr = map_.find(ghostIter->id());
630  if (ptr == 0) {
631  UTIL_THROW("find(ghostIter->id()) == 0");
632  }
633  // We do NOT test if ptr == ghostIter.get(), because it is possible
634  // to have multiple atoms with the same id on one processor. One to
635  // one correspondence of ids and pointers is guaranteed only for
636  // local atoms.
637  }
638  #if 0
639  if (j != nGhost()) {
640  UTIL_THROW("Number counted by ghostIterator != nGhost()");
641  }
642  #endif
643 
644  return true;
645  }
