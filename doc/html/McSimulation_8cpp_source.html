128  void McSimulation::setOptions(int argc, char **argv)
129  {
130  bool eflag = false; // echo
131  bool rFlag = false; // restart file
132  bool pFlag = false; // param file
133  bool cFlag = false; // command file
134  bool iFlag = false; // input prefix
135  bool oFlag = false; // output prefix
136  #ifdef MCMD_PERTURB
137  bool fflag = false; // free energy perturbation
138  #endif
139  char* rarg = 0;
140  char* pArg = 0;
141  char* cArg = 0;
142  char* iArg = 0;
143  char* oArg = 0;
144 
145  // Read program arguments
146  int c;
147  opterr = 0;
148  while ((c = getopt(argc, argv, "er:p:c:i:o:f")) != -1) {
149  switch (c) {
150  case 'e':
151  eflag = true;
152  break;
153  case 'r':
154  rFlag = true;
155  rarg = optarg;
156  break;
157  case 'p': // parameter file
158  pFlag = true;
159  pArg = optarg;
160  break;
161  case 'c': // command file
162  cFlag = true;
163  cArg = optarg;
164  break;
165  case 'i': // input prefix
166  iFlag = true;
167  iArg = optarg;
168  break;
169  case 'o': // output prefix
170  oFlag = true;
171  oArg = optarg;
172  break;
173  #ifdef MCMD_PERTURB
174  case 'f':
175  fflag = true;
176  break;
177  #endif
178  case '?':
179  Log::file() << "Unknown option -" << optopt << std::endl;
180  UTIL_THROW("Invalid command line option");
181  }
182  }
183 
184  // Set flag to echo parameters as they are read.
185  if (eflag) {
186  Util::ParamComponent::setEcho(true);
187  }
188 
189  #ifdef MCMD_PERTURB
190  // Set to use a perturbation.
191  if (fflag) {
192 
193  if (rFlag) {
194  std::string msg("Error: Options -r and -p are incompatible. Use -r alone. ");
195  msg += "Existence of a perturbation is specified in restart file.";
196  UTIL_THROW(msg.c_str());
197  }
198 
199  // Set to expect perturbation in the param file.
200  system().setExpectPerturbation();
201 
202  #ifdef UTIL_MPI
203  Util::Log::file() << "Set to read parameters from a single file"
204  << std::endl;
205  setIoCommunicator();
206  #endif
207 
208  }
209  #endif
210 
211  // If option -p, set parameter file name
212  if (pFlag) {
213  if (rFlag) {
214  UTIL_THROW("Cannot have both parameter and restart files");
215  }
216  fileMaster().setParamFileName(std::string(pArg));
217  }
218 
219  // If option -c, set command file name
220  if (cFlag) {
221  fileMaster().setCommandFileName(std::string(cArg));
222  }
223 
224  // If option -i, set path prefix for input files
225  if (iFlag) {
226  fileMaster().setInputPrefix(std::string(iArg));
227  }
228 
229  // If option -o, set path prefix for output files
230  if (oFlag) {
231  fileMaster().setOutputPrefix(std::string(oArg));
232  }
233 
234  // If option -r, restart
235  if (rFlag) {
236  //Log::file() << "Reading restart file "
237  // << std::string(rarg) << std::endl;
238  isRestarting_ = true;
239  load(std::string(rarg));
240  }

390  /*
391  * Read and execute commands from a specified command file.
392  */
393  void McSimulation::readCommands(std::istream &in)
394  {
395  if (!isInitialized_) {
396  UTIL_THROW("McSimulation is not initialized");
397  }
398 
399  std::string command;
400  std::string filename;
401  std::ifstream inputFile;
402  std::ofstream outputFile;
403 
404  #ifndef UTIL_MPI
405  std::istream& inBuffer = in;
406  #else
407  std::stringstream inBuffer;
408  std::string line;
409  #endif
410 
411  bool readNext = true;
412  while (readNext) {
413 
414  // iffdef UTIL_MPI, read a line and copy to stringstream inBuffer
415  // ifndef UTIL_MPI, inBuffer is simply a reference to istream in.
416 
417  #ifdef UTIL_MPI
418  // Read a command line, and broadcast if necessary.
419  if (!hasIoCommunicator() || isIoProcessor()) {
420  getNextLine(in, line);
421  }
422  if (hasIoCommunicator()) {
423  bcast<std::string>(communicator(), line, 0);
424  }
425 
426  // Copy the command line to inBuffer.
427  inBuffer.clear();
428  for (unsigned i=0; i < line.size(); ++i) {
429  inBuffer.put(line[i]);
430  }
431  #endif
432 
433  inBuffer >> command;
434  Log::file() << command;
435 
436  if (isRestarting_) {
437 
438  if (command == "RESTART") {
439  int endStep;
440  inBuffer >> endStep;
441  Log::file() << " " << iStep_ << " to "
442  << endStep << std::endl;
443  simulate(endStep, isRestarting_);
444  isRestarting_ = false;
445  } else {
446  UTIL_THROW("Missing RESTART command");
447  }
448 
449  } else {
450 
451  if (command == "FINISH") {
452  Log::file() << std::endl;
453  readNext = false;
454  } else
455  if (command == "SET_CONFIG_IO") {
456  std::string classname;
457  inBuffer >> classname;
458  Log::file() << Str(classname, 15) << std::endl;
459  system().setConfigIo(classname);
460  } else
461  if (command == "READ_CONFIG") {
462  inBuffer >> filename;
463  Log::file() << Str(filename, 15) << std::endl;
464  fileMaster().openInputFile(filename, inputFile);
465  system().readConfig(inputFile);
466  inputFile.close();
467  } else
468  if (command == "SIMULATE") {
469  int endStep;
470  inBuffer >> endStep;
471  Log::file() << " " << endStep << std::endl;
472  bool isContinuation = false;
473  simulate(endStep, isContinuation);
474  } else
475  if (command == "CONTINUE") {
476  if (iStep_ == 0) {
477  UTIL_THROW("Attempt to continue when iStep_ == 0");
478  }
479  int endStep;
480  inBuffer >> endStep;
481  Log::file() << Int(endStep, 15) << std::endl;
482  bool isContinuation = true;
483  simulate(endStep, isContinuation);
484  } else
485  if (command == "ANALYZE_CONFIGS") {
486  int min, max;
487  inBuffer >> min >> max >> filename;
488  Log::file() << " " << min << " " << max
489  << " " << filename << std::endl;
490  analyzeConfigs(min, max, filename);
491  } else
492  if (command == "ANALYZE_TRAJECTORY") {
493  std::string classname;
494  std::string filename;
495  int min, max;
496  inBuffer >> min >> max >> classname >> filename;
497  Log::file() << " " << Str(classname,15)
498  << " " << Str(filename, 15)
499  << std::endl;
500  analyzeTrajectory(min, max, classname, filename);
501  } else
502  if (command == "WRITE_CONFIG") {
503  inBuffer >> filename;
504  Log::file() << Str(filename, 15) << std::endl;
505  fileMaster().openOutputFile(filename, outputFile);
506  system().writeConfig(outputFile);
507  outputFile.close();
508  } else
509  if (command == "WRITE_PARAM") {
510  inBuffer >> filename;
511  Log::file() << " " << filename << std::endl;
512  fileMaster().openOutputFile(filename, outputFile);
513  writeParam(outputFile);
514  outputFile.close();
515  } else
516  if (command == "GENERATE_MOLECULES") {
517  DArray<double> ExclusionRadii;
518  DArray<int> capacities;
519  ExclusionRadii.allocate(nAtomType());
520  capacities.allocate(nSpecies());
521 
522  // Parse command
523  inBuffer >> system().boundary();
524  Log::file() << " " << system().boundary();
525  Label capacityLabel("Capacities:");
526  inBuffer >> capacityLabel;
527  for (int iSpecies = 0; iSpecies < nSpecies(); ++iSpecies) {
528  inBuffer >> capacities[iSpecies];
529  Log::file() << " " << capacities[iSpecies];
530  }
531  Label radiusLabel("Radii:");
532  inBuffer >> radiusLabel;
533  for (int iType=0; iType < nAtomType(); iType++) {
534  inBuffer >> ExclusionRadii[iType];
535  Log::file() << " " << ExclusionRadii[iType];
536  }
537  Log::file() << std::endl;
538 
539  for (int iSpecies = 0; iSpecies < nSpecies(); ++iSpecies) {
540  species(iSpecies).generateMolecules(
541  capacities[iSpecies], ExclusionRadii, system(),
542  #ifdef INTER_BOND
543  &system().bondPotential(),
544  #endif
545  system().boundary());
546  }
547 
548  #ifndef INTER_NOPAIR
549  // Generate cell list
550  system().pairPotential().buildCellList();
551  #endif
552 
553  } else
554  if (command == "DEFORM_CELL") {
555 
556  // Read in configuration from file
557  inBuffer >> filename;
558  Log::file() << Str(filename, 15) << std::endl;
559  fileMaster().openInputFile(filename, inputFile);
560  system().readConfig(inputFile);
561  inputFile.close();
562 
563  System::MoleculeIterator molIter;
564  Molecule::AtomIterator atomIter;
565  for (int iSpec=0; iSpec < nSpecies(); ++iSpec) {
566  system().begin(iSpec, molIter);
567  for ( ; molIter.notEnd(); ++molIter) {
568  molIter->begin(atomIter);
569  for ( ; atomIter.notEnd(); ++atomIter) {
570  Vector cartPosition, genPosition;
571  cartPosition = atomIter->position();
572  system().boundary().transformCartToGen(cartPosition, genPosition);
573  atomIter->position() = genPosition;
574  }
575  }
576  }
577 
578  // Read in new boundary
579  inBuffer >> system().boundary();
580  Log::file() << " " << system().boundary();
581  Log::file() << std::endl;
582 
583  for (int iSpec=0; iSpec < nSpecies(); ++iSpec) {
584  system().begin(iSpec, molIter);
585  for ( ; molIter.notEnd(); ++molIter) {
586  molIter->begin(atomIter);
587  for ( ; atomIter.notEnd(); ++atomIter) {
588  Vector cartPosition, genPosition;
589  genPosition = atomIter->position();
590  system().boundary().transformGenToCart(genPosition, cartPosition);
591  atomIter->position() = cartPosition;
592  }
593  }
594  }
595 
596  // Write out configuration to file
597  inBuffer >> filename;
598  Log::file() << Str(filename, 15) << std::endl;
599  fileMaster().openOutputFile(filename, outputFile);
600  system().writeConfig(outputFile);
601  outputFile.close();
602 
603  #ifndef INTER_NOPAIR
604  // Generate cell list
605  system().pairPotential().buildCellList();
606  #endif
607 
608  } else
609  #ifndef UTIL_MPI
610  #ifndef INTER_NOPAIR
611  if (command == "SET_PAIR") {
612  std::string paramName;
613  int typeId1, typeId2;
614  double value;
615  inBuffer >> paramName >> typeId1 >> typeId2 >> value;
616  Log::file() << " " << paramName
617  << " " << typeId1 << " " << typeId2
618  << " " << value << std::endl;
619  system().pairPotential()
620  .set(paramName, typeId1, typeId2, value);
621  } else
622  #endif
623  #ifdef INTER_BOND
624  if (command == "SET_BOND") {
625  std::string paramName;
626  int typeId;
627  double value;
628  inBuffer >> paramName >> typeId >> value;
629  Log::file() << " " << paramName << " " << typeId
630  << " " << value << std::endl;
631  system().bondPotential().set(paramName, typeId, value);
632  } else
633  #endif
634  #ifdef INTER_ANGLE
635  if (command == "SET_ANGLE") {
636  std::string paramName;
637  int typeId;
638  double value;
639  inBuffer >> paramName >> typeId >> value;
640  Log::file() << " " << paramName << " " << typeId
641  << " " << value << std::endl;
642  system().anglePotential().set(paramName, typeId, value);
643  } else
644  #endif
645  #ifdef INTER_DIHEDRAL
646  if (command == "SET_DIHEDRAL") {
647  std::string paramName;
648  int typeId;
649  double value;
650  inBuffer >> paramName >> typeId >> value;
651  Log::file() << " " << paramName << " " << typeId
652  << " " << value << std::endl;
653  system().dihedralPotential().set(paramName, typeId, value);
654  } else
655  #endif // ifdef INTER_DIHEDRAL
656  #endif // ifndef UTIL_MPI
657  {
658  Log::file() << " Error: Unknown command " << std::endl;
659  readNext = false;
660  }
661 
662  }
663  }

678  /*
679  * Run this MC simulation.
680  */
681  void McSimulation::simulate(int endStep, bool isContinuation)
682  {
683  if (!isInitialized_) {
684  UTIL_THROW("McSimulation not initialized");
685  }
686 
687  if (isContinuation) {
688  Log::file() << "Restarting from iStep = "
689  << iStep_ << std::endl;
690  } else {
691  iStep_ = 0;
692  analyzerManager().setup();
693  mcMoveManagerPtr_->setup();
694  }
695  int beginStep = iStep_;
696  int nStep = endStep - beginStep;
697  Log::file() << std::endl;
698 
699  // Main Monte Carlo loop
700  Timer timer;
701  timer.start();
702  for ( ; iStep_ < endStep; ++iStep_) {
703 
704  // Analyzers and restart outut
705  if (Analyzer::baseInterval > 0) {
706  if (iStep_ % Analyzer::baseInterval == 0) {
707  save(saveFileName_);
708  analyzerManager().sample(iStep_);
709  }
710  }
711 
712  // Choose and attempt an McMove
713  mcMoveManagerPtr_->chooseMove().move();
714 
715  #ifdef UTIL_MPI
716  #ifdef MCMD_PERTURB
717  // Attempt replica move, if any.
718  if (system().hasPerturbation()) {
719  if (system().hasReplicaMove()) {
720  if (system().replicaMove().isAtInterval(iStep_)) {
721  bool success = system().replicaMove().move();
722  #ifndef INTER_NOPAIR
723  if (success) {
724  system().pairPotential().buildCellList();
725  }
726  #endif
727  }
728  }
729  }
730  #endif
731  #endif
732 
733  }
734  timer.stop();
735  double time = timer.time();
736 
737  // Final analyzers / save
738  assert(iStep_ == endStep);
739  if (Analyzer::baseInterval > 0) {
740  if (iStep_ % Analyzer::baseInterval == 0) {
741  save(saveFileName_);
742  analyzerManager().sample(iStep_);
743  }
744  }
745 
746  // Output results of all analyzers to output files
747  analyzerManager().output();
748 
749  // Output results of move statistics to files
750  mcMoveManagerPtr_->output();
751 
752  // Output time for the run
753  Log::file() << std::endl;
754  Log::file() << "endStep " << endStep << std::endl;
755  Log::file() << "nStep " << nStep << std::endl;
756  Log::file() << "run time " << time
757  << " sec" << std::endl;
758  double rStep = double(nStep);
759  Log::file() << "time / nStep " << time / rStep
760  << " sec" << std::endl;
761  Log::file() << std::endl;
762 
763  // Print McMove acceptance statistics
764  long attempt;
