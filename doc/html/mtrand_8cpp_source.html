74  * Generate new state vector.
75  */
76  void MTRand_int32::gen_state()
77  {
78  for (int i = 0; i < (n - m); ++i)
79  state[i] = state[i + m] ^ twiddle(state[i], state[i + 1]);
80  for (int i = n - m; i < (n - 1); ++i)
81  state[i] = state[i + m - n] ^ twiddle(state[i], state[i + 1]);
82  state[n - 1] = state[m - 1] ^ twiddle(state[n - 1], state[0]);
83  p = 0; // reset position
84  }

89  void MTRand_int32::seed(unsigned long s) {
90  state[0] = s & 0xFFFFFFFFUL; // for > 32 bit machines
91  for (int i = 1; i < n; ++i) {
92  state[i] = 1812433253UL * (state[i - 1] ^ (state[i - 1] >> 30)) + i;
93  // see Knuth TAOCP Vol2. 3rd Ed. P.106 for multiplier
94  // in the previous versions, MSBs of the seed affect only MSBs of the array state
95  // 2002/01/09 modified by Makoto Matsumoto
96  state[i] &= 0xFFFFFFFFUL; // for > 32 bit machines
97  }
98  p = n; // force gen_state() to be called for next random number
99  }

101  void MTRand_int32::seed(const unsigned long* array, int size) { // init by array
102  seed(19650218UL);
103  int i = 1, j = 0;
104  for (int k = ((n > size) ? n : size); k; --k) {
105  state[i] = (state[i] ^ ((state[i - 1] ^ (state[i - 1] >> 30)) * 1664525UL))
106  + array[j] + j; // non linear
107  state[i] &= 0xFFFFFFFFUL; // for > 32 bit machines
108  ++j; j %= size;
109  if ((++i) == n) { state[0] = state[n - 1]; i = 1; }
110  }
111  for (int k = n - 1; k; --k) {
112  state[i] = (state[i] ^ ((state[i - 1] ^ (state[i - 1] >> 30)) * 1566083941UL)) - i;
113  state[i] &= 0xFFFFFFFFUL; // for > 32 bit machines
114  if ((++i) == n) { state[0] = state[n - 1]; i = 1; }
115  }
116  state[0] = 0x80000000UL; // MSB is 1; assuring non-zero initial array
117  p = n; // force gen_state() to be called for next random number
