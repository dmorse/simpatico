152  /*
153  * Evaluate end-to-end vectors of all chains, add to ensemble.
154  */
155  void ComMSD::sample(long iStep)
156  {
157  if (!isAtInterval(iStep)) return;
158 
159  // Confirm that nMolecule has remained constant, and nMolecule > 0.
160  if (nMolecule_ <= 0) {
161  UTIL_THROW("nMolecule <= 0");
162  }
163  if (nMolecule_ != system().nMolecule(speciesId_)) {
164  UTIL_THROW("Number of molecules has changed.");
165  }
166 
167  Vector r;
168  IntVector shift;
169  Vector lengths = system().boundary().lengths();
170  int i, j;
171  for (i = 0; i < nMolecule_; ++i) {
172 
173  r = system().molecule(speciesId_, i).atom(0).position();
174  system().boundary().shift(r);
175 
176  // Compare current r to previous position, oldPositions_[i]
177  system().boundary().distanceSq(r, oldPositions_[i], shift);
178 
179  // If atom 0 crossed a boundary, increment its shift vector
180  shifts_[i] += shift;
181 
182  // Store current position in box for comparison to next one
183  oldPositions_[i] = r;
184 
185  // Add the other particle's positions to C.O.M.
186  moleculeCom(i, r);
187 
188  // Reconstruct true position
189  for (j = 0; j < Dimension; ++j) {
190  truePositions_[i][j] = r[j] + double(shifts_[i][j])*lengths[j];
191  }
192 
193  }
194  accumulator_.sample(truePositions_);
195 
196  }

225  /*
226  * Evaluate center of mass of a single molecule, given a starting vector R0.
227  */
228  void ComMSD::moleculeCom(int iMol, Vector &R0)
229  {
230  Vector R = R0, r1, r2, dr;
231 
232  // Assuming a linear chain (including Ring molecules).
233  for (int i = 1; i < nAtom_-1; ++i) {
234  r1 = system().molecule(speciesId_, iMol).atom(i-1).position();
235  r2 = system().molecule(speciesId_, iMol).atom(i).position();
236  system().boundary().distanceSq(r2, r1, dr);
237  R += dr;
238  R0 += R;
239  }
240 
241  // Do the average.
242  R0 /= double(nAtom_);
243  }
