114  /*
115  * Construct grid of cells, build linked list and identify neighbors.
116  */
117  void CellList::makeGrid(const Vector& lower, const Vector& upper, const Vector& cutoffs)
118  {
119 
120  // Calculate required grid dimensions
121  setGridDimensions(lower, upper, cutoffs);
122 
123  if (grid_.size() > cells_.capacity()) {
124  UTIL_THROW("Insufficient memory was allocated for this grid");
125  }
126 
127  // Mark all cells to as ghost cells by default.
128  int ic;
129  for (ic = 0; ic < grid_.size(); ++ic) {
130  cells_[ic].setIsGhostCell(true);
131  }
132 
133  // Build linked list of local cells.
134  IntVector p;
135  Cell* prevPtr = 0;
136  Cell* cellPtr = 0;
137  for (p[0] = 1; p[0] < grid_.dimension(0) - 1; ++p[0]) {
138  for (p[1] = 1; p[1] < grid_.dimension(1) - 1; ++p[1]) {
139  for (p[2] = 1; p[2] < grid_.dimension(2) - 1; ++p[2]) {
140  ic = grid_.rank(p);
141  cellPtr = &cells_[ic];
142  cellPtr->setIsGhostCell(false);
143  if (prevPtr) {
144  prevPtr->setNextCell(*cellPtr);
145  } else {
146  begin_ = cellPtr;
147  }
148  prevPtr = cellPtr;
149  }
150  }
151  }
152  cellPtr->setLastCell();
153 
154  // Calculate range of displacements to neighbor cells
155  IntVector dmin, dmax;
156  for (int i = 0; i < Dimension; ++i) {
157  if (grid_.dimension(i) > 2) {
158  dmin[i] = -1;
159  dmax[i] = 1;
160  } else if (grid_.dimension(i) == 2) {
161  dmin[i] = 0;
162  dmax[i] = 1;
163  } else if (grid_.dimension(i) == 1) {
164  dmin[i] = 0;
165  dmax[i] = 0;
166  }
167  }
168 
169  // Construct array of integer offsets to neighbors
170  IntVector span;
171  IntVector d;
172  int offset;
173  span[2] = 1;
174  span[1] = grid_.dimension(2);
175  span[0] = grid_.dimension(2)*grid_.dimension(1);
176  offsets_.clear();
177  offsets_.append(0);
178  for (d[0] = dmin[0]; d[0] <= dmax[0]; ++d[0]) {
179  for (d[1] = dmin[1]; d[1] <= dmax[1]; ++d[1]) {
180  for (d[2] = dmin[2]; d[2] <= dmax[2]; ++d[2]) {
181  offset = d[0]*span[0] + d[1]*span[1] + d[2];
182  if (offset != 0) {
183  offsets_.append(offset);
184  }
185  }
186  }
187  }
188 
189  }

295  /*
296  * Check validity of CellList, throw an Exception if an error is found.
297  */
298  bool CellList::isValid() const
299  {
300 
301  if (isAllocated()) {
302  if (tags_.capacity() <= 0) {
303  UTIL_THROW("CellList is allocated but tags_.capacity() <= 0");
304  }
305  if (handles_.capacity() <= 0) {
306  UTIL_THROW("CellList is allocated but handles_.capacity() <= 0");
307  }
308  }
309 
310  if (isBuilt_) {
311 
312  if (!isAllocated()) {
313  UTIL_THROW("CellList is built but not allocated");
314  }
315 
316  // Check validity of all cells individually.
317  const Atom* atomPtr;
318  const Cell* cellPtr;
319  int nAtomCell;
320  int nAtomSum = 0;
321  for (int icell = 0; icell < grid_.size(); ++icell) {
322  cellPtr = &cells_[icell];
323  nAtomCell = cellPtr->nAtom();
324  if (nAtomCell != cellPtr->atomCapacity()) {
325  UTIL_THROW("Cell nAtom != atomCapacity");
326  }
327  if (nAtomCell > 0) {
328  for (int i = 0; i < nAtomCell; ++i) {
329  atomPtr = cellPtr->atomPtr(i);
330  if (atomPtr == 0)
331  UTIL_THROW("Null Atom* in a Cell");
332  if (icell != cellIndexFromPosition(atomPtr->position())) {
333  UTIL_THROW("Inconsistent position");
334  }
335  }
336  }
337  nAtomSum += nAtomCell;
338  }
339 
340  // Check that total number of atoms in all cells equals nAtom.
341  // Note: nAtom is incremented by the placeAtom() method.
342  if (nAtom_ >= 0) {
343  if (nAtomSum != nAtom_) {
344  UTIL_THROW("Number of atoms in all cells != nAtom");
345  }
346  }
347 
348  } else { // if not isBuilt_
349 
350  if (nAtom_ != 0) {
351  UTIL_THROW("CellList is not built, but nAtom_ != 0");
352  }
353 
354  }
355 
356  return true;
357  }
