93  * Initialize tables for spring constants and normalizations.
94  */
95  void CfbRebridgeBase::setup()
96  {
97  double l20;
98  double lmin, lmax, del, pi;
99  int i;
100 
101  pi = acos(-1.0);
102 
103  // Range of l_20; 2.0 is empirical.
104  lmax = 2.0 * (length10_ + length21_);
105 
106  // Fill table elements for: l_20 <= lmin
107  if (length21_ == length10_) {
108  // 0.1 is empirical.
109  lmin = length10_ / length21_ / sqrt(kappa10_) * 0.1;
110  l20Table[0] = lmin;
111  angTable[0] = pi/2.0;
112  kappaTable[0] = 0.0;
113  normalTable[0] = 2.0;
114  } else {
115  // 0.5 is empirical.
116  lmin = fabs(length10_ - length21_) * 0.5;
117  l20Table[0] = lmin;
118  orientationBiasTable(lmin, angTable[0], kappaTable[0], normalTable[0]);
119  }
120 
121  // Fill table elements for: lmin < l_20 < lmax
122  del = (lmax - lmin) / double(MaxBin_-2);
123  l20 = lmin - del/2.0;
124  for (i=1; i<MaxBin_-1; i++) {
125  l20 += del;
126  l20Table[i] = l20;
127  orientationBiasTable(l20, angTable[i], kappaTable[i], normalTable[i]);
128  }
129 
130  // Fill table elements for: l_20 >= lmax
131  l20Table[MaxBin_-1] = lmax;
132  angTable[MaxBin_-1] = 0.0;
133  kappaTable[MaxBin_-1] = kappa10_ * length21_ * lmax;
134  computeNormalization(angTable[MaxBin_-1], kappaTable[MaxBin_-1],
135  normalTable[MaxBin_-1]);

138  /*
139  * Configuration bias algorithm for deleting the last particle of an interior
140  * bridge.
141  *
142  * 1
143  * o-----o 0
144  * /
145  * o
146  * 2
147  * 2 = prev, 0 = next, 1 = crank-shaft rotating
148  *
149  * Note that in the current implementation, the end atom position will be
150  * modified, so the calling subroutine is responsible for recovering the old
151  * position if necessary. This is similar to the implementation of
152  * CfbEndBase::deleteEndAtom.
153  */
154  void CfbRebridgeBase::deleteMiddleAtom(

156  int prevBType, int nextBType,
157  double &rosenbluth, double &energy)
158  {
159  Vector prevPos = prevPtr->position();
160  Vector nextPos = nextPtr->position();

163  double trialEnergy, bondEnergy, wExt, length, lengthSq;
164  double l_20, l_21, bias;
165  double prefAng, kappaAng, normConst;
166  int iTrial;
167  bool ready;
168 
169  // Bond length and bonding vector 2 --> 0
170  lengthSq = boundary().distanceSq(nextPos, prevPos, u_20);
171  l_20 = sqrt(lengthSq);
172  u_20 /= l_20;
173 
174  // Bond length and bonding vector 2 --> 1 & bonding energy
175  lengthSq = boundary().distanceSq(partPos, prevPos, u_21);
176  l_21 = sqrt(lengthSq);
177  u_21 /= l_21;
178  energy = system().bondPotential().energy(lengthSq, prevBType);
179 
180  // Bond length between 1 and 0 & bonding energy
181  lengthSq = boundary().distanceSq(partPos, nextPos);
182  bondEnergy = system().bondPotential().energy(lengthSq, nextBType);
183  energy += bondEnergy;
184 
185  // Nonbonded pair energy of current position
186  #ifndef INTER_NOPAIR
187  trialEnergy = system().pairPotential().atomEnergy(*partPtr);
188  #else
189  trialEnergy = 0.0;
190  #endif
191 
192  #ifdef INTER_ANGLE

194  trialEnergy += system().anglePotential().atomEnergy(*partPtr);
195  }
196  #endif
197 
198  #ifdef INTER_EXTERNAL
199  if (system().hasExternalPotential()) {
200  trialEnergy += system().externalPotential().atomEnergy(*partPtr);
201  }
202  #endif
203 
204  energy += trialEnergy;
205

207  getAngKappaNorm(l_20, prefAng, kappaAng, normConst);
208  orientationBias(u_21, u_20, prefAng, kappaAng, bias);
209

211  rosenbluth = normConst / bias;
212 
213  // start to accumulate Rosenbluth factor
214  wExt = boltzmann(trialEnergy + bondEnergy);
215

217  length = l_21;
218  for (iTrial=0; iTrial < nTrial_ - 1; iTrial++) {
219  ready = false;
220  while (!ready) {
221  random().unitVector(bondVec);
222  bondVec *= length;
223  trialPos.add(prevPos, bondVec);
224  boundary().shift(trialPos);
225 
226  lengthSq = boundary().distanceSq(trialPos, prevPos, u_21);
227  l_21 = sqrt(lengthSq);
228  u_21 /= l_21;
229  orientationBias(u_21, u_20, prefAng, kappaAng, bias);
230 
231  // Rejection method
232  if (bias > random().uniform(0.0, 1.0)) {
233  ready = true;
234  }
235  }
236 
237  // Bond 1-0 potential energy
238  lengthSq = boundary().distanceSq(trialPos, nextPos);
239  bondEnergy = system().bondPotential().energy(lengthSq, nextBType);
240  partPtr->position() = trialPos;
241  #ifndef INTER_NOPAIR
242  trialEnergy = system().pairPotential().atomEnergy(*partPtr);
243  #else
244  trialEnergy = 0.0;
245  #endif

256  }
257  #endif

263  rosenbluth *= wExt;

269  * bridge.

271  void CfbRebridgeBase::addMiddleAtom(
272  Atom* partPtr, Atom* prevPtr, Atom* nextPtr,
273  int prevBType, int nextBType,
274  double &rosenbluth, double &energy)
275  {
276  Vector prevPos = prevPtr->position();
277  Vector nextPos = nextPtr->position();
278  Vector u_20, u_21, bondVec;
279  Vector trialPos[MaxTrial_];
280  double trialEnergy[MaxTrial_], bondEnergy[MaxTrial_];
281  double trialProb[MaxTrial_], bias[MaxTrial_];
282  double l_20, l_21;
283  double beta, length, lengthSq, wExt;
284  double prefAng, kappaAng, normConst;
285  int iTrial;
286  bool ready;
287 
288  // Bond 2 --> 0
289  lengthSq = boundary().distanceSq(nextPos, prevPos, u_20);
290  l_20 = sqrt(lengthSq);
291  u_20 /= l_20;
292 
293  // Bond 2-1 & bonding energy
294  beta = energyEnsemble().beta();
295  length = system().bondPotential().
296  randomBondLength(&random(), beta, prevBType);
297  l_21 = length;
298  energy = system().bondPotential().energy(l_21*l_21, prevBType);
299 
300  // Loop over trial orientations
301  wExt = 0.0;
302 
303  // Compute normalization factor for the orientation bias.
304  getAngKappaNorm(l_20, prefAng, kappaAng, normConst);
305 
306  // Generating trial orientations
307  for (iTrial=0; iTrial < nTrial_; iTrial++) {
308  ready = false;
309  while (!ready) {
310  random().unitVector(bondVec);
311  bondVec *= length;
312  trialPos[iTrial].add(prevPos, bondVec);
313  boundary().shift(trialPos[iTrial]);
314 
315  lengthSq = boundary().distanceSq(trialPos[iTrial], prevPos, u_21);
316  l_21 = sqrt(lengthSq);
317  u_21 /= l_21;
318  orientationBias(u_21, u_20, prefAng, kappaAng, bias[iTrial]);
319 
320  // Rejection method
321  if (bias[iTrial] > random().uniform(0.0, 1.0)) {
322  ready = true;
323  }
324  }
325 
326  // Bond 1-0 potential energy
327  lengthSq = boundary().distanceSq(trialPos[iTrial], nextPos);
328  bondEnergy[iTrial] =
329  system().bondPotential().energy(lengthSq, nextBType);
330  partPtr->position() = trialPos[iTrial];
331 
332  #ifndef INTER_NOPAIR
333  trialEnergy[iTrial] = system().pairPotential().atomEnergy(*partPtr);
334  #else
335  trialEnergy[iTrial] = 0.0;
336  #endif
337 
338  #ifdef INTER_ANGLE
339  if (system().hasAnglePotential()) {
340  trialEnergy[iTrial] +=
341  system().anglePotential().atomEnergy(*partPtr);
342  }
343  #endif
344 
345  #ifdef INTER_EXTERNAL
346  if (system().hasExternalPotential()) {
347  trialEnergy[iTrial] +=
348  system().externalPotential().atomEnergy(*partPtr);
349  }
350  #endif
351 
352  trialProb[iTrial] =
353  boltzmann(trialEnergy[iTrial] + bondEnergy[iTrial]);
354  wExt += trialProb[iTrial];
355  }
356 
357  // Normalize trial probabilities
358  for (iTrial=0; iTrial < nTrial_; iTrial++) {
359  trialProb[iTrial] = trialProb[iTrial] / wExt;
360  }
361 
362  // Choose trial position
363  iTrial = random().drawFrom(trialProb, nTrial_);
364 
365  // Add non-bonded pair energy for chosen trial
366  energy += trialEnergy[iTrial];
367 
368  // Add 1-0 bond energy
369  energy += bondEnergy[iTrial];
370 
371  // Copy trial position to newPos
372  partPtr->position() = trialPos[iTrial];
373 
374  // update Rosenbluth weight and remove the orientation bias
375  rosenbluth = wExt * normConst / bias[iTrial];
376  }

484  /*
485  * First determine the preferential orientation angle and angle biasing
486  * spring constant; then compute the statistical weight for the biasing
487  * function using Simpson's rule:
488  *
489  * Note: the variou branches represent certain geometrical extreme
490  * situations.
491  */
492  void CfbRebridgeBase::orientationBiasTable
493  (double l_20, double &prefAng, double &kappaAng, double &normConst)
494  {
495  double cos_ang;
496 
497  // Preferential orientation & effective spring constant
498  if ( l_20 <= fabs(length21_ - length10_) ) {
499 
500  if (length21_ > length10_) {
501  prefAng = 0.0;
502  kappaAng = kappa10_ * (length21_ - l_20 - length10_);
503  kappaAng *= length21_ * l_20 / (length21_ - l_20);
504  } else if (length21_ == length10_) {
505  prefAng = acos(-1.0)/2.0;
506  kappaAng = 0.0;
507  } else {
508  prefAng = acos(-1.0);
509  kappaAng = kappa10_ * (length10_ - l_20 - length21_);
510  kappaAng *= length21_ * l_20 / (length21_ + l_20);
511  }
512 
513  } else if ( l_20 >= (length21_ + length10_) ) {
514 
515  prefAng = 0.0;
516  kappaAng = kappa10_ * (l_20 - length21_ - length10_) * length21_ * l_20
517  / (l_20 - length21_);
518 
519  } else {
520 
521  cos_ang = (l_20*l_20 + length21_*length21_ - length10_*length10_)
522  * 0.5 / length21_ / l_20;
523  prefAng = acos(cos_ang);
524  kappaAng = length21_ * l_20 / length10_;
525  kappaAng = kappa10_ * kappaAng * kappaAng * (1.0 - cos_ang*cos_ang);
526 
527  }
528 
529  // Calculate normalization factor.
530  if (kappaAng < 1.0E-7) {
531  normConst = 2.0;
532  } else {
533  computeNormalization(prefAng, kappaAng, normConst);
534  }
535  }
536 
537  /*
538  * Use Simpson's rule to compute the normalization factor
539  * int { exp(-0.5*beta*kappaAng*(x-prefAng)**2)*sin(x) , {x, 0, pi} }
540  */
541  void CfbRebridgeBase::computeNormalization
542  (double prefAng, double kappaAng, double &normConst)
543  {
544  int i, N; // N must be even
545  double delta, x, sum, tmp;
546 
547  // indices for Simpson's rule: 0,1,...,N
548  N = 400; // default value
549  tmp = 0.5*kappaAng;
550  delta = acos(-1.0)/double(N);
551 
552  // the end points contribute nothing since sin(0)=sin(pi)=0.
553  sum = 0.0;
554  x = -delta;
555  for (i=0; i < N-1; i=i+2) {
556  x += delta;
557  //sum += 2.0 * exp( -tmp * (x-prefAng)*(x-prefAng) ) * sin(x);
558  sum += 2.0 * boltzmann( tmp * (x-prefAng)*(x-prefAng) ) * sin(x);
559  x += delta;
560  //sum += 4.0 * exp( -tmp * (x-prefAng)*(x-prefAng) ) * sin(x);
561  sum += 4.0 * boltzmann( tmp * (x-prefAng)*(x-prefAng) ) * sin(x);
562  }
563  normConst = delta*sum/3.0;
564  }
