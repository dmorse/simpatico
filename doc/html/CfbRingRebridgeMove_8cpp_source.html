98  /*
99  * Generate, attempt and accept or reject a Monte Carlo move.
100  *
101  * Convention for index
102  * nAtom
103  * _____________________________________
104  * | |
105  * 0 1 2 3 ... nAtom-1
106  * o---o---o---o---o---o---o---o---o---o
107  * bonds: 0 1 2 3 nAtom-1
108  *
109  * The algorithm is slightly different from that for linear molecules since a
110  * ring molecule does not have ends. The operation of moving the atom pointers
111  * have to account for the periodic boundary conditions, which is achived with
112  * the help of modId(int s, int n) method.
113  *
114  */
115  bool CfbRingRebridgeMove::move()
116  {
117  Molecule *molPtr; // pointer to randomly chosen molecule
118  Atom *thisPtr; // pointer to the current end atom
119  Atom *tempPtr; // pointer to the current end atom
120  int *bonds, iBond;
121  int nAtom, sign, beginId, endId, i;
122  double rosen_r, rosen_f;
123  double energy_r, energy_f;
124  bool accept;
125 
126  incrementNAttempt();
127 
128  // Choose a molecule at random
129  molPtr = &(system().randomMolecule(speciesId_));
130  nAtom = molPtr->nAtom();
131 
132  // Require that chain length - 2 >= nRegrow_
133  if (nRegrow_ > nAtom - 2) {
134  UTIL_THROW("nRegrow_ >= chain length");
135  }
136 
137  // Allocate bonds array
138  bonds = new int[nRegrow_ + 1];
139 
140  // Choose the beginId of the growing interior bridge
141  beginId = random().uniformInt(0, nAtom);
142 
143  // Choose direction and fill bonds array: sign = +1 if beginId < endId
144  if (random().uniform(0.0, 1.0) > 0.5) {
145  sign = +1;
146  endId = (beginId + (nRegrow_ - 1)) % nAtom;
147 
148  // Save bond types
149  iBond = endId;
150  for (i = 0; i < nRegrow_ + 1; ++i) {
151  bonds[i] = molPtr->bond(iBond).typeId();
152  iBond--;
153  if (iBond < 0) iBond += nAtom;
154  }
155  } else {
156  sign = -1;
157  endId = beginId;
158  beginId = (endId + (nRegrow_ - 1)) % nAtom;
159 
160  // Save bond types
161  iBond = endId - 1;
162  if (iBond < 0) iBond += nAtom;
163  for (i = 0; i < nRegrow_ + 1; ++i) {
164  bonds[i] = molPtr->bond(iBond).typeId();
165  iBond++;
166  if (iBond >= nAtom) iBond -= nAtom;
167  }
168  }
169 
170  // Store current atomic positions from segment to be regrown
171  tempPtr = &(molPtr->atom(beginId));
172  for (i = 0; i < nRegrow_; ++i) {
173  thisPtr = tempPtr + modId(beginId + i*sign, nAtom) - beginId;
174  oldPos_[i] = thisPtr->position();
175  }
176 
177  // Delete atoms: endId -> beginId
178  deleteSequence(sign, molPtr, endId, bonds, rosen_r, energy_r);
179 
180  // Regrow atoms: endId -> beginId
181  addSequence(sign, molPtr, beginId, bonds, rosen_f, energy_f);
182 
183  // Release bonds array
184  delete [] bonds;
185 
186  // Decide whether to accept or reject
187  accept = random().metropolis(rosen_f/rosen_r);
188  if (accept) {
189 
190  // Increment counter for accepted moves of this class.
191  incrementNAccept();
192 
193  // If the move is accepted, keep current positions.
194 
195  } else {
196 
197  // If the move is rejected, restore old positions
198  tempPtr = &(molPtr->atom(beginId));
199  for (i = 0; i < nRegrow_; ++i) {
200  thisPtr = tempPtr + modId(beginId + i*sign, nAtom) - beginId;
201  //system().moveAtom(*thisPtr, oldPos_[i]);
202  thisPtr->position() = oldPos_[i];
203  #ifndef INTER_NOPAIR
204  system().pairPotential().updateAtomCell(*thisPtr);
205  #endif
206  }
207 
208  }
209 
210  return accept;
211  }

214  * Delete a consecutive sequence of atoms in a Ring.
215  */
216  void CfbRingRebridgeMove::deleteSequence(int sign, Molecule *molPtr,
217  int endId, int *bonds, double &rosenbluth, double &energy)
218  {
219  Atom *endPtr;
220  Atom *thisPtr;
221  Atom *prevPtr;
222  Atom *nextPtr;
223  int prevBType, nextBType, nAtom;
224  double rosen_r, energy_r;
225 
226  // Initialize weight
227  rosenbluth = 1.0;
228  energy = 0.0;
229 
230  nAtom = molPtr->nAtom();
231  endPtr = &(molPtr->atom(endId));
232 
233  // Step 1: delete the last atom
234  if (nRegrow_ >= 1) {
235  thisPtr = endPtr;
236  prevPtr = endPtr + modId(endId - sign,nAtom) - endId;
237  nextPtr = endPtr + modId(endId + sign,nAtom) - endId;
238 
239  nextBType = bonds[0];
240  prevBType = bonds[1];
241 
242  // Orientation biased trimer rebridge move
243  deleteMiddleAtom(thisPtr, prevPtr, nextPtr,
244  prevBType, nextBType, rosen_r, energy_r);
245  #ifndef INTER_NOPAIR
246  system().pairPotential().deleteAtom(*thisPtr);
247  #endif
248  rosenbluth *= rosen_r;
249  energy += energy_r;
250  }
251 
252  // step 2: delete the remaining atoms
253  for (int i = 0; i < nRegrow_ - 1; ++i) {
254  thisPtr = endPtr + modId(endId - (i+1)*sign, nAtom) - endId;
255  prevPtr = endPtr + modId(endId - (i+2)*sign, nAtom) - endId;
256 
257  prevBType = bonds[i+2];
258  deleteEndAtom(thisPtr, prevPtr, prevBType, rosen_r, energy_r);
259  #ifndef INTER_NOPAIR
260  system().pairPotential().deleteAtom(*thisPtr);
261  #endif
262 
263  rosenbluth *= rosen_r;
264  energy += energy_r;
265  }
266  }

271  void CfbRingRebridgeMove::addSequence(int sign, Molecule *molPtr,
272  int beginId, int *bonds, double &rosenbluth, double &energy)
273  {

279  double rosen_f, energy_f;
280 
281  // Initialize weight
282  rosenbluth = 1.0;
283  energy = 0.0;
284 
285  // Step 1: grow the first nRegrow_ - 1 atoms
286  nAtom = molPtr->nAtom();
287  beginPtr = &(molPtr->atom(beginId));
288 
289  for (int i = 0; i < nRegrow_ - 1; ++i) {
290  thisPtr = beginPtr + modId(beginId+i*sign,nAtom) - beginId;
291  prevPtr = beginPtr + modId(beginId+(i-1)*sign,nAtom) - beginId;
292 
293  prevBType = bonds[nRegrow_ - i];
294  addEndAtom(thisPtr, prevPtr, prevBType, rosen_f, energy_f);
295  #ifndef INTER_NOPAIR
296  system().pairPotential().addAtom(*thisPtr);
297  #endif
298 
299  rosenbluth *= rosen_f;
300  energy += energy_f;
301  }
302 
303  // Step 2: grow the last atoms
304  if (nRegrow_ >= 1) {
305  prevPtr = beginPtr + modId(beginId+(nRegrow_-2)*sign,nAtom) - beginId;
306  thisPtr = beginPtr + modId(beginId+(nRegrow_-1)*sign,nAtom) - beginId;
307  nextPtr = beginPtr + modId(beginId+nRegrow_*sign,nAtom) - beginId;
308 
309  prevBType = bonds[1];
310  nextBType = bonds[0];
311 
312  // Invoke the orientation biased trimer re-bridging move
313  addMiddleAtom(thisPtr, prevPtr, nextPtr,
314  prevBType, nextBType, rosen_f, energy_f);
315  #ifndef INTER_NOPAIR
316  system().pairPotential().addAtom(*thisPtr);
317  #endif
318 
319  rosenbluth *= rosen_f;
320  energy += energy_f;
321  }
322  }
