100  * Generate, attempt and accept or reject a Monte Carlo move.
101  *
102  * Convention for index
103  * head of molecule tail of molecule
104  * | |
105  * 0 1 2 3 ... length-1
106  * o---o---o---o---o---o---o---o---o---o
107  * |_______________|___________|
108  * valid beginId nRegrow
109  *
110  */
111  bool CfbRebridgeMove::move()
112  {
113  Molecule *molPtr; // pointer to randomly chosen molecule
114  Atom *thisPtr; // pointer to the current end atom
115  int length, sign, beginId, endId, i;
116  int *bonds;
117  double rosen_r, rosen_f;
118  double energy_r, energy_f;
119  bool accept;
120 
121  incrementNAttempt();
122 
123  // Choose a molecule at random
124  molPtr = &(system().randomMolecule(speciesId_));
125  length = molPtr->nAtom();
126 
127  // Require that chain length - 2 >= nRegrow_
128  if (nRegrow_ > length - 2) {
129  UTIL_THROW("nRegrow_ >= chain length");
130  }
131 
132  // Allocate bonds array
133  bonds = new int[nRegrow_ + 1];
134 
135  // Choose the beginId of the growing interior bridge
136  beginId = random().uniformInt(1, length - nRegrow_);
137 
138  // Choose direction: sign = +1 if beginId < endId
139  if (random().uniform(0.0, 1.0) > 0.5) {
140  sign = +1;
141  endId = beginId + (nRegrow_ - 1);
142  } else {
143  sign = -1;
144  endId = beginId;
145  beginId = endId + (nRegrow_ - 1);
146  }
147 
148  // Store current atomic positions from segment to be regrown
149  thisPtr = &(molPtr->atom(beginId));
150  for (i = 0; i < nRegrow_; ++i) {
151  oldPos_[i] = thisPtr->position();
152  thisPtr += sign;
153  }
154 
155  // Get bond types.
156  if (sign == +1) {
157  for (i = 0; i < nRegrow_ + 1; ++i)
158  bonds[i] = molPtr->bond(endId - i).typeId();
159  } else {
160  for (i = 0; i < nRegrow_ + 1; ++i)
161  bonds[i] = molPtr->bond(endId - 1 + i).typeId();
162  }
163 
164  // Delete atoms: endId -> beginId
165  thisPtr = &(molPtr->atom(endId));
166  deleteSequence(nRegrow_, sign, thisPtr, bonds, rosen_r, energy_r);
167 
168  // Regrow atoms: endId -> beginId
169  thisPtr = &(molPtr->atom(beginId));
170  addSequence(nRegrow_, sign, thisPtr, bonds, rosen_f, energy_f);
171 
172  // Release bonds array
173  delete [] bonds;
174 
175  // Decide whether to accept or reject
176  accept = random().metropolis(rosen_f/rosen_r);
177  if (accept) {
178 
179  // Increment counter for accepted moves of this class.
180  incrementNAccept();
181 
182  // If the move is accepted, keep current positions.
183 
184  } else {
185 
186  // If the move is rejected, restore old positions
187  thisPtr = &(molPtr->atom(beginId));
188  for (i = 0; i < nRegrow_; ++i) {
189  //system().moveAtom(*thisPtr, oldPos_[i]);
190  thisPtr->position() = oldPos_[i];
191  #ifndef INTER_NOPAIR
192  system().pairPotential().updateAtomCell(*thisPtr);
193  #endif
194  thisPtr += sign;
195  }
196 
197  }
