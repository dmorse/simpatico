66  * Read frame, return false if end-of-file
67  */
68  bool LammpsDumpReader::readFrame()
69  {
70  // Preconditions
71  if (!positions_.isAllocated()) {
72  UTIL_THROW("positions_ array is not allocated");
73  }
74 
75  bool notEnd;
76  std::stringstream line;
77 
78  // Attempt to read first line
79  notEnd = getNextLine(file_, line);
80  if (!notEnd) {
81  return false;
82  }
83 
84  // Process ITEM: TIMESTEP
85  checkString(line, "ITEM:");
86  checkString(line, "TIMESTEP");
87  int step;
88  file_ >> step;
89 
90  // Read ITEM: NUMBER OF ATOMS
91  notEnd = getNextLine(file_, line);
92  if (!notEnd) {
93  UTIL_THROW("EOF reading ITEM: NUMBER OF ATOMS");
94  }
95  checkString(line, "ITEM:");
96  checkString(line, "NUMBER");
97  checkString(line, "OF");
98  checkString(line, "ATOMS");
99  int nAtom;
100  file_ >> nAtom;
101  if (nAtom != nAtomTotal_) {
102  UTIL_THROW("Inconsistent values: nAtom != nAtomTotal_");
103  }
104 
105  // Read ITEM: BOX
106  notEnd = getNextLine(file_, line);
107  if (!notEnd) {
108  UTIL_THROW("EOF reading ITEM: BOX");
109  }
110  checkString(line, "ITEM:");
111  checkString(line, "BOX");
112  // Ignore rest of ITEM: BOX line
113  Vector min, max, lengths;
114  for (int i = 0; i < Dimension; ++i) {
115  file_ >> min[i] >> max[i];
116  lengths[i] = max[i] - min[i];
117  }
118  boundary().setOrthorhombic(lengths);
119 
120  // Read ITEM: ATOMS
121  notEnd = getNextLine(file_, line);
122  checkString(line, "ITEM:");
123  checkString(line, "ATOMS");
124  // Ignore the rest of ITEM: ATOMS line, for now
125 
126  // Load positions into positions_ vector, indexed by id.
127  IntVector shift;
128  int id, typeId, molId, i, j;
129  for (i = 0; i < nAtom; ++i) {
130 
131  file_ >> id;
132  id = id - 1; // Convert convention, Lammps -> Simpatico
133  file_ >> typeId;
134  file_ >> molId;
135 
136  // Read position
137  for (j = 0; j < Util::Dimension; ++j) {
138  file_ >> positions_[id][j];
139  }
140  file_ >> shift;
141 
142  // shift into simulation cell
143  boundary().shift(positions_[id]);
144 
145  }
146 
147  // Assign atom positions, assuming ordered atom ids
148  int iSpecies, iMol;
149  Species *speciesPtr;
150  Molecule::AtomIterator atomIter;
151  Molecule *molPtr;
152  id = 0;
153  for (iSpecies = 0; iSpecies < simulation().nSpecies(); ++iSpecies) {
154  speciesPtr = &simulation().species(iSpecies);
155  for (iMol = 0; iMol < speciesPtr->capacity(); ++iMol) {
156  molPtr = &system().molecule(iSpecies, iMol);
157  for (molPtr->begin(atomIter); atomIter.notEnd(); ++atomIter) {
158  atomIter->position() = positions_[id];
159  id++;
160  }
161  }
162  }
163 
164  return true;
165  }
