127  rFlag = true;

130  case 'p': // parameter file

133  break;
134  case 'c': // command file

153  UTIL_THROW("Invalid command line option");
154  }
155  }
156 
157  // Set flag to echo parameters as they are read.
158  if (eflag) {
159  Util::ParamComponent::setEcho(true);
160  }
161 
162  #ifdef MCMD_PERTURB
163  // Set to use a perturbation.
164  if (fflag) {
165

304  /*
305  * Read and implement commands in an input script.
306  */
307  void MdSimulation::readCommands(std::istream &in)
308  {
309  std::string command;
310  std::string filename;
311  std::ifstream inputFile;
312  std::ofstream outputFile;
313 
314  #ifndef UTIL_MPI
315  std::istream& inBuffer = in;
316  #else
317  std::stringstream inBuffer;
318  std::string line;
319  #endif
320 
321  bool readNext = true;
322  while (readNext) {
323 
324  #ifdef UTIL_MPI
325  if (!hasIoCommunicator() || isIoProcessor()) {
326  getNextLine(in, line);
327  }
328  if (hasIoCommunicator()) {
329  bcast<std::string>(communicator(), line, 0);
330  }
331  inBuffer.clear();
332  for (unsigned i=0; i < line.size(); ++i) {
333  inBuffer.put(line[i]);
334  }
335  #endif
336 
337  inBuffer >> command;
338  Log::file().setf(std::ios_base::left);
339  Log::file().width(15);
340  Log::file() << command;
341 
342  if (isRestarting_) {
343 
344  if (command == "RESTART") {
345  int endStep;
346  inBuffer >> endStep;
347  Log::file() << " " << iStep_ << " to "
348  << endStep << std::endl;
349  simulate(endStep, isRestarting_);
350  isRestarting_ = false;
351  } else {
352  UTIL_THROW("Missing RESTART command");
353  }
354 
355  } else {
356 
357  if (command == "FINISH") {
358  Log::file() << std::endl;
359  readNext = false;
360  } else
361  if (command == "READ_CONFIG") {
362  inBuffer >> filename;
363  Log::file() << Str(filename, 15) << std::endl;
364  fileMaster().openInputFile(filename, inputFile);
365  system().readConfig(inputFile);
366  inputFile.close();
367  } else
368  if (command == "THERMALIZE") {
369  double temperature;
370  inBuffer >> temperature;
371  Log::file() << Dbl(temperature, 15, 6) << std::endl;
372  system().setBoltzmannVelocities(temperature);
373  system().removeDriftVelocity();
374  } else
375  if (command == "SIMULATE") {
376  int endStep;
377  inBuffer >> endStep;
378  Log::file() << Int(endStep, 15) << std::endl;
379  bool isContinuation = false;
380  simulate(endStep, isContinuation);
381  } else
382  if (command == "CONTINUE") {
383  if (iStep_ == 0) {
384  UTIL_THROW("Attempt to continue simulation when iStep_ == 0");
385  }
386  int endStep;
387  inBuffer >> endStep;
388  Log::file() << Int(endStep, 15) << std::endl;
389  bool isContinuation = true;
390  simulate(endStep, isContinuation);
391  } else
392  if (command == "ANALYZE_CONFIGS") {
393  int min, max;
394  inBuffer >> min >> max >> filename;
395  Log::file() << Int(min, 15) << Int(max, 15)
396  << Str(filename, 20) << std::endl;
397  analyzeConfigs(min, max, filename);
398  } else
399  if (command == "WRITE_CONFIG") {
400  inBuffer >> filename;
401  Log::file() << Str(filename, 15) << std::endl;
402  fileMaster().openOutputFile(filename, outputFile);
403  system().writeConfig(outputFile);
404  outputFile.close();
405  } else
406  if (command == "WRITE_PARAM") {
407  inBuffer >> filename;
408  Log::file() << Str(filename, 15) << std::endl;
409  fileMaster().openOutputFile(filename, outputFile);
410  writeParam(outputFile);
411  outputFile.close();
412  } else
413  if (command == "SET_CONFIG_IO") {
414  std::string classname;
415  inBuffer >> classname;
416  Log::file() << Str(classname, 15) << std::endl;
417  system().setConfigIo(classname);
418  } else
419  if (command == "ANALYZE_TRAJECTORY") {
420  std::string classname;
421  std::string filename;
422  int min, max;
423  inBuffer >> min >> max >> classname >> filename;
424  Log::file() << " " << Str(classname,15)
425  << " " << Str(filename, 15)
426  << std::endl;
427  analyzeTrajectory(min, max, classname, filename);
428  } else
429  if (command == "GENERATE_MOLECULES") {
430  DArray<double> ExclusionRadii;
431  DArray<int> capacities;
432  ExclusionRadii.allocate(nAtomType());
433  capacities.allocate(nSpecies());
434 
435  // Parse command
436  inBuffer >> system().boundary();
437  Log::file() << " " << system().boundary();
438  Label capacityLabel("Capacities:");
439  inBuffer >> capacityLabel;
440  for (int iSpecies = 0; iSpecies < nSpecies(); ++iSpecies) {
441  inBuffer >> capacities[iSpecies];
442  Log::file() << " " << capacities[iSpecies];
443  }
444  Label radiusLabel("Radii:");
445  inBuffer >> radiusLabel;
446  for (int iType=0; iType < nAtomType(); iType++) {
447  inBuffer >> ExclusionRadii[iType];
448  Log::file() << " " << ExclusionRadii[iType];
449  }
450  Log::file() << std::endl;
451 
452  for (int iSpecies = 0; iSpecies < nSpecies(); ++iSpecies) {
453  species(iSpecies).generateMolecules(
454  capacities[iSpecies], ExclusionRadii, system(),
455  #ifdef INTER_BOND
456  &system().bondPotential(),
457  #endif
458  system().boundary());
459  }
460 
461  #ifndef INTER_NOPAIR
462  // Generate pair list
463  system().pairPotential().buildPairList();
464  #endif
465 
466  } else
467  #ifndef UTIL_MPI
468  #ifndef INTER_NOPAIR
469  if (command == "SET_PAIR") {
470  std::string paramName;
471  int typeId1, typeId2;
472  double value;
473  inBuffer >> paramName >> typeId1 >> typeId2 >> value;
474  Log::file() << " " << paramName
475  << " " << typeId1 << " " << typeId2
476  << " " << value << std::endl;
477  system().pairPotential()
478  .set(paramName, typeId1, typeId2, value);
479  } else
480  #endif
481  #ifdef INTER_BOND
482  if (command == "SET_BOND") {
483  std::string paramName;
484  int typeId;
485  double value;
486  inBuffer >> paramName >> typeId >> value;
487  Log::file() << " " << paramName << " " << typeId
488  << " " << value << std::endl;
489  system().bondPotential().set(paramName, typeId, value);
490  } else
491  #endif
492  #ifdef INTER_ANGLE
493  if (command == "SET_ANGLE") {
494  std::string paramName;
495  int typeId;
496  double value;
497  inBuffer >> paramName >> typeId >> value;
498  Log::file() << " " << paramName << " " << typeId
499  << " " << value << std::endl;
500  system().anglePotential().set(paramName, typeId, value);
501  } else
502  #endif
503  #ifdef INTER_DIHEDRAL
504  if (command == "SET_DIHEDRAL") {
505  std::string paramName;
506  int typeId;
507  double value;
508  inBuffer >> paramName >> typeId >> value;
509  Log::file() << " " << paramName << " " << typeId
510  << " " << value << std::endl;
511  system().dihedralPotential().set(paramName, typeId, value);
512  } else
513  #endif
514  #endif // ifndef UTIL_MPI
515  {
516  Log::file() << "Error: Unknown command " << std::endl;
517  readNext = false;
518  }
519 
520  }
521  }

535  /*
536  * Run a simulation.
537  */
538  void MdSimulation::simulate(int endStep, bool isContinuation)
539  {
540  Timer timer;
541 
542  if (isContinuation) {
543  Log::file() << "Continuing simulation from iStep = "
544  << iStep_ << std::endl;
545  } else {
546  iStep_ = 0;
547  system().shiftAtoms();
548  system().calculateForces();
549  analyzerManager().setup();
550  system_.mdIntegrator().setup();
551  }
552  int beginStep = iStep_;
553  int nStep = endStep - beginStep;
554 
555  // Main loop
556  Log::file() << std::endl;
557  timer.start();
558  for ( ; iStep_ < endStep; ++iStep_) {
559 
560  // Sample scheduled analyzers
561  if (Analyzer::baseInterval > 0) {
562  if (iStep_ % Analyzer::baseInterval == 0) {
563  system().shiftAtoms();
564  analyzerManager().sample(iStep_);
565  }
566  }
567  if (saveInterval_ > 0) {
568  if (iStep_ % saveInterval_ == 0) {
569  save(saveFileName_);
570  }
571  }
572 
573  #ifdef INTER_NOPAIR
574  else {
575  // When the pair potential is disabled, require that
576  // Analyzer::baseInterval != 0 to guarantee periodic
577  // shifting of atomic positions into primary cell.
578  UTIL_THROW("Analyzer::baseInterval == 0 with no pair potential");
579  }
580  #endif
581 
582  // Take one MD step with the MdIntegrator
583  system_.mdIntegrator().step();
584 
585  }
586  timer.stop();
587  double time = timer.time();
588  double rstep = double(nStep);
589 
590  // Shift final atomic positions
591  system().shiftAtoms();
592 
593  // Final analyzers and restart
594  if (Analyzer::baseInterval > 0) {
595  if (iStep_ % Analyzer::baseInterval == 0) {
596  analyzerManager().sample(iStep_);
597  }
598  }
599  if (saveInterval_ > 0) {
600  if (iStep_ % saveInterval_ == 0) {
601  save(saveFileName_);
602  }
603  }
604 
605  // Final analyzer output
606  analyzerManager().output();
607 
608  // Output timing results
609  Log::file() << std::endl;
610  Log::file() << "Time Statistics" << std::endl;
611  Log::file() << "endStep " << endStep << std::endl;
612  Log::file() << "nStep " << nStep << std::endl;
613  Log::file() << "run time "
614  << time << " sec" << std::endl;
615  Log::file() << "time / nStep "
616  << time / rstep << " sec" << std::endl;
617  Log::file() << "time / (nStep*nAtom) "
618  << time / (rstep*double(system().nAtom()))
619  << " sec" << std::endl;
620  Log::file() << std::endl;
621  Log::file() << std::endl;
622 
623  #ifndef INTER_NOPAIR
624  Log::file() << "PairList Statistics" << std::endl;
625  Log::file() << "maxNPair "
626  << system().pairPotential().pairList().maxNPair()
627  << std::endl;
628  Log::file() << "maxNAtom "
629  << system().pairPotential().pairList().maxNAtom()
630  << std::endl;
631  int buildCounter = system().pairPotential().pairList().buildCounter();
632  Log::file() << "buildCounter "
633  << buildCounter
634  << std::endl;
635  Log::file() << "steps / build "
636  << rstep/double(buildCounter)
637  << std::endl;
638  Log::file() << std::endl;
639  #endif
640 
641  }

647  MdSimulation::analyzeConfigs(int min, int max, std::string basename)
648  {
649  // Preconditions
650  if (min < 0) UTIL_THROW("min < 0");
651  if (max < min) UTIL_THROW("max < min");
652 
653  Timer timer;
654  std::string filename;
655  std::stringstream indexString;
656  std::ifstream configFile;
657  int nConfig;
658  nConfig = max - min + 1;
659 
660  // Main loop
661  Log::file() << "begin main loop" << std::endl;
662  timer.start();
663  for (iStep_ = min; iStep_ <= max; ++iStep_) {
664 
665  indexString << iStep_;
666  filename = basename;
667  filename += indexString.str();
668  fileMaster().openInputFile(filename, configFile);
669 
670  // Clear the stringstream
671  indexString.str("");
672 
673  system().readConfig(configFile);
674  configFile.close();
675 
676  #ifndef INTER_NOPAIR
677  // Build the system CellList
678  system().pairPotential().buildPairList();
679  #endif
680 
681  #ifndef NDEBUG
682  isValid();
683  #endif
684 
685  // Initialize analyzers (taking in molecular information).
686  if (iStep_ == min) analyzerManager().setup();
687 
688  // Sample property values
689  analyzerManager().sample(iStep_);
690 
691  // Clear out the System for the next readConfig.
692  system().removeAllMolecules();
693 
694  }
695  timer.stop();
696  Log::file()<< "end main loop" << std::endl;
697 
698  // Output results of all analyzers to output files
699  analyzerManager().output();
700 
701  // Output time
702  Log::file() << std::endl;
703  Log::file() << "nConfig " << nConfig << std::endl;
704  Log::file() << "run time " << timer.time()
705  << " sec" << std::endl;
706  Log::file() << "time / config " << timer.time()/double(nConfig)
707  << " sec" << std::endl;
708  Log::file() << std::endl;
709 
710  }

712  /*
713  * Open, read and analyze a trajectory file
714  */
715  void MdSimulation::analyzeTrajectory(int min, int max,
716  std::string classname,
717  std::string filename)
718  {
719  // Preconditions
720  if (min < 0) UTIL_THROW("min < 0");
721  if (max < 0) UTIL_THROW("max < 0");
722  if (max < min) UTIL_THROW("max < min!");
723 
724  // Construct TrajectoryReader
725  TrajectoryReader* trajectoryReaderPtr;
726  trajectoryReaderPtr
727  = system().trajectoryReaderFactory().factory(classname);
728  if (!trajectoryReaderPtr) {
729  std::string message;
730  message = "Invalid TrajectoryReader class name " + classname;
731  UTIL_THROW(message.c_str());
732  }
733 
734  // Open trajectory file
735  Log::file() << "Reading " << filename << std::endl;
736  trajectoryReaderPtr->open(filename);
737 
738  // Main loop over trajectory frames
739  Timer timer;
740  Log::file() << "Begin main loop" << std::endl;
741  bool hasFrame = true;
742  timer.start();
743  for (iStep_ = 0; iStep_ <= max && hasFrame; ++iStep_) {
744  hasFrame = trajectoryReaderPtr->readFrame();
745  if (hasFrame) {
746  #ifndef INTER_NOPAIR
747  // Build the system PairList
748  system().pairPotential().buildPairList();
749  #endif
750  #ifdef UTIL_DEBUG
751  isValid();
752  #endif
753  // Initialize analyzers (taking in molecular information).
754  if (iStep_ == min) analyzerManager().setup();
755  // Sample property values only for iStep >= min
756  if (iStep_ >= min) analyzerManager().sample(iStep_);
757  }
758  }
759  timer.stop();
760  Log::file() << "end main loop" << std::endl;
761  int nFrames = iStep_ - min;
762 
763  trajectoryReaderPtr->close();
764  delete trajectoryReaderPtr;
