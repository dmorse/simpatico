121  /*
122  * Initialize data - called by readParameters and loadParameters (private).
123  */
124  void Domain::initialize()
125  {
126  // Validate gridDimensions
127  int nproc = 1;
128  for (int i = 0; i < Dimension; i++) {
129  if (gridDimensions_[i] <= 0) {
130  UTIL_THROW("Processor grid dimensions must be greater than 0");
131  }
132  nproc *= gridDimensions_[i];
133  }
134  int commSize = 1;
135  #ifdef UTIL_MPI
136  commSize = intracommPtr_->Get_size();
137  #endif
138  if (nproc != commSize) {
139  UTIL_THROW("Grid dimensions inconsistent with communicator size");
140  }
141 
142  // Set grid dimensions
143  grid_.setDimensions(gridDimensions_);
144 
145  // Mark all directions as periodic.
146  for (int i = 0; i < Dimension; i++) {
147  gridIsPeriodic_[i] = true;
148  }
149 
150  // Find grid coordinates for this processor
151  gridCoordinates_ = grid_.position(gridRank_);
152 
153  IntVector sourceCoordinates;
154  int i, j, k, jp;
155 
156  // Loop over Cartesian direction for send
157  for (i = 0; i < Dimension; ++i) {
158 
159  // Loop over increment sign (+1 or -1 grid coordinate i)
160  for (j = 0; j < 2; ++j) {
161 
162  // Increment to source along coordinate i
163  k = (j == 0 ? 1 : -1);
164 
165  // Complementary index (other direction)
166  jp = (j == 0 ? 1 : 0);
167 
168  //Initialize coordinates of the partner processor
169  sourceCoordinates = gridCoordinates_;
170  sourceCoordinates[i] = gridCoordinates_[i] + k;
171 
172  // Check for transfer past boundary
173  if (gridIsPeriodic_[i] == true) {
174  shift_(i, j) = grid_.shift(sourceCoordinates[i], i);
175  sourceRanks_(i, j) = grid_.rank(sourceCoordinates);
176  destRanks_(i, jp) = sourceRanks_(i, j);
177  } else {
178  shift_(i, j) = 0;
179  if (!grid_.isInGrid(sourceCoordinates[i], i)) {
180  destRanks_(i, j) = -1;
181  sourceRanks_(i, jp) = -1;
182  }
183  }
184 
185  }
186  }
187  isInitialized_ = true;
188  }
