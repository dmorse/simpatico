66  readOutputFileName(in);

87  }

93  truePositions_.allocate(speciesCapacity);
94  oldPositions_.allocate(speciesCapacity);
95  shifts_.allocate(speciesCapacity);

98  accumulator_.setParam(speciesCapacity, capacity_);

104  * Initialize at beginning of simulation.
105  */
106  void AtomMSD::setup()
107  {
108  if (!isInitialized_) {
109  UTIL_THROW("Error: object is not initialized");
110  }
111  Species& species = configuration().species(speciesId_);
112  Boundary& boundary = configuration().boundary();
113 
114  // Set number of molecules of this species in the Configuration.
115  nMolecule_ = species.size();
116  accumulator_.setNEnsemble(nMolecule_);
117  accumulator_.clear();
118 
119  // Store initial positions, and set initial shift vectors.
120  Vector r;
121  IntVector zero(0);

123  int i = 0;

125  species.begin(iter);

127  r = iter->atom(atomId_).position;

135  /*
136  * Evaluate end-to-end vectors of all chains, add to ensemble.
137  */
138  void AtomMSD::sample(long iStep)
139  {
140  if (isAtInterval(iStep)) {
141  Species& species = configuration().species(speciesId_);
142  Boundary& boundary = configuration().boundary();
143 
144  // Confirm that nMolecule is positive, and unchanged.
145  if (nMolecule_ <= 0) {
146  UTIL_THROW("nMolecule <= 0");
147  }
148  if (nMolecule_ != species.size()) {
149  UTIL_THROW("Number of molecules has changed.");
150  }
151 
152  Vector r;
153  IntVector shift;
154  Vector lengths = boundary.lengths();
155  int i, j;
156  Species::Iterator iter;
157 
158  i = 0;
159  for (species.begin(iter); iter.notEnd(); ++iter) {
160 
161  r = iter->atom(atomId_).position;
162  boundary.shift(r);
163 
164  // Compare current r to previous position, oldPositions_[i]
165  boundary.distanceSq(r, oldPositions_[i], shift);
166 
167  // If a boundary was crossed, increment shift vector
168  shifts_[i] += shift;
169 
170  // Reconstruct true position
171  for (j = 0; j < Dimension; ++j) {
172  truePositions_[i][j] = r[j] + shifts_[i][j]*lengths[j];
173  }
174 
175  // Store current position in box for comparison to next one
176  oldPositions_[i] = r;
177 
178  ++i;
179  }
180  accumulator_.sample(truePositions_);
181  }
