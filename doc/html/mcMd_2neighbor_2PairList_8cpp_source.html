164  /*
165  * Build the PairList, i.e., populate it with atom pairs.
166  */
167  void PairList::build(const Boundary& boundary)
168  {
169  CellList::NeighborArray cellNeighbor;
170  Vector iPos, jPos;
171  Atom *iAtomPtr, *jAtomPtr;
172  double dRSq, cutoffSq;
173  int nCellNeighbor, nCellAtom, totCells;
174  int ic, ip, iAtomId, jp, jAtomId;
175  bool foundNeighbor;
176 
177  // Precondition
178  assert(isAllocated());
179 
180  // Set maximum squared-separation for pairs in Pairlist
181  cutoffSq = cutoff_*cutoff_;
182 
183  // Initialize counters for primary atoms and neighbors
184  nAtom1_ = 0; // Number of primary atoms with neighbors
185  nAtom2_ = 0; // Number of secondary atoms (2nd in pair), or pairs
186  nAtom_ = 0; // Number of atoms, with or without neighbors
187  tList1_ = atomCapacity_ - 1;
188  first_[0] = 0;
189 
190  // Loop over cells containing primary atom. ic = cell index
191  totCells = cellList_.totCells();
192  for (ic = 0; ic < totCells; ++ic) {
193 
194  // Get Array cellNeighbor of Ids of neighbor atoms for cell ic.
195  // Elements 0,..., nCellAtom - 1 contain Ids for atoms in cell ic.
196  // Elements nCellAtom,..., nCellNeighbor-1 are from neighboring cells.
197  cellList_.getCellNeighbors(ic, cellNeighbor, nCellAtom);
198  nCellNeighbor = cellNeighbor.size();
199 
200  // Loop over atoms in cell ic
201  for (ip = 0; ip < nCellAtom; ++ip) {
202  iAtomPtr = cellNeighbor[ip];
203  iPos = iAtomPtr->position();
204  iAtomId = iAtomPtr->id();
205  ++nAtom_;
206  if (nAtom_ > atomCapacity_) {
207  UTIL_THROW("Overflow: nAtom_ > atomCapacity_ in PairList");
208  }
209 
210  // Indicate that no neighbors have been found yet
211  foundNeighbor = false;
212 
213  // Loop over atoms in all neighboring cells, including cell ic.
214  for (jp = 0; jp < nCellNeighbor; ++jp) {
215  jAtomPtr = cellNeighbor[jp];
216  jPos = jAtomPtr->position();
217  jAtomId = jAtomPtr->id();
218 
219  // Avoid double counting: only count pairs with jAtomId > iAtomId
220  if ( jAtomId > iAtomId ) {
221 
222  // Exclude bonded pairs
223  if (!iAtomPtr->mask().isMasked(*jAtomPtr)) {
224 
225  // Calculate distance between atoms i and j
226  dRSq = boundary.distanceSq(iPos, jPos);
227 
228  if (dRSq < cutoffSq) {
229 
230  // Check for overflow of atom2Ptrs_
231  if (nAtom2_ >= pairCapacity_) {
232  UTIL_THROW("Overflow: # pairs > pairCapacity_");
233  }
234 
235  // If first_ neighbor, record iAtomId in atom1Ptrs_
236  if (!foundNeighbor) {
237 
238  assert(nAtom1_ < atomCapacity_);
239  assert(nAtom1_ >= 0);
240  assert(tList1_ >= nAtom1_);
241 
242  atom1Ptrs_[nAtom1_] = iAtomPtr;
243  oldPositions_[nAtom1_] = iPos;
244  foundNeighbor = true;
245  }
246 
247  // Append Id of 2nd atom in pair to atom2Ptrs_[]
248  assert(tList1_ < atomCapacity_);
249  atom2Ptrs_[nAtom2_] = jAtomPtr;
250  ++nAtom2_;
251 
252  }
253  }
254 
255  } // end if jAtomId > iAtomId
256 
257  } // end for jp (j atom)
258 
259  if (foundNeighbor) {
260  // Update counter first_[nAtom1_]
261  ++nAtom1_;
262  first_[nAtom1_] = nAtom2_;
263  } else {
264  // Add atom with no neighbors to top of atom1Ptrs_
265  assert(tList1_ >= nAtom1_);
266  assert(tList1_ < atomCapacity_);
267  assert(tList1_ >= 0);
268  atom1Ptrs_[tList1_] = iAtomPtr;
269  oldPositions_[tList1_] = iPos;
270  --tList1_;
271  }
272 
273  } // end for ip (i atom)
274 
275  } // end for ic (i cell)
276 
277  // Increment buildCounter_= number of times the list has been built.
278  ++buildCounter_;
279 
280  // Increment maximum
281  if (nAtom_ > maxNAtom_) maxNAtom_ = nAtom_;
282  if (nAtom2_ > maxNAtom2_) maxNAtom2_ = nAtom2_;
283 
284  }

301  /*
302  * Return false if any atom has moved a distance greater than skin/2,
303  * or if the PairList has never been built. Return true otherwise.
304  */
305  bool PairList::isCurrent(const Boundary& boundary) const
306  {
307  // Precondition
308  assert(isAllocated());
309 
310  double dRSq, dRSqMax;
311  int ip;
312 
313  // If the list has never been built, it is not current.
314  if (buildCounter_ == 0) return false;
315 
316  dRSqMax = 0.25*skin_*skin_;
317 
318  // Loop over atoms with neighbors (increment from bottom)
319  for (ip = 0; ip < nAtom1_; ++ip) {
320  dRSq = boundary.distanceSq(atom1Ptrs_[ip]->position(),
321  oldPositions_[ip]);
322  if (dRSq > dRSqMax) {
323  return false;
324  }
325  }
326 
327  // Loop over atoms with no neighbors (decrement from top)
328  for (ip = atomCapacity_ - 1; ip > tList1_; --ip) {
329  dRSq = boundary.distanceSq(atom1Ptrs_[ip]->position(),
330  oldPositions_[ip]);
331  if (dRSq > dRSqMax) {
332  return false;
333  }
334  }
335 
336  return true;
337  }

382  }
