48  void CfbReptationMove::readParameters(std::istream& in)
49  {
50  // Read parameters
51  readProbability(in);
52  read<int>(in, "speciesId", speciesId_);
53  read<int>(in, "nTrial", nTrial_);
54 
55  // Check that 0 < nTrial_ <= MaxTrial
56  if (nTrial_ <=0 || nTrial_ > MaxTrial_) {
57  UTIL_THROW("Invalid value input for nTrial");
58  }
59 
60  // Check that the Species is a subclass of Linear
61  Linear* speciesPtr;
62  speciesPtr = dynamic_cast<Linear*>(&(simulation().species(speciesId_)));
63  if (!speciesPtr) {
64  UTIL_THROW("Species is not a subclass of Linear");
65  }
66 
67  // Identify bond type
68  bondTypeId_ = speciesPtr->speciesBond(0).typeId();
69 
70  // Check that bond type is the same for all bonds
71  int nBond = speciesPtr->nBond();
72  for (int i = 1; i < nBond; ++i) {
73  if (bondTypeId_ != speciesPtr->speciesBond(i).typeId()) {
74  UTIL_THROW("Unequal bond type ids");
75  }
76  }
77 
78  // Identify policy for masking nonbonded interactions.
79  maskPolicy_ = simulation().maskedPairPolicy();
80 
81  // Allocate memory for junction arrays
82  int nAtom = speciesPtr->nAtom();
83  junctions_.allocate(nAtom-1);
84  lTypes_.allocate(nAtom-1);
85  uTypes_.allocate(nAtom-1);
86 
87  // Identify junctions between atoms of different types
88  int lType, uType;
89  nJunction_ = 0;
90  for (int i = 0; i < nAtom - 1; ++i) {
91  lType = speciesPtr->atomTypeId(i);
92  uType = speciesPtr->atomTypeId(i+1);
93  if (lType != uType) {
94  junctions_[nJunction_] = i;
95  lTypes_[nJunction_] = lType;
96  uTypes_[nJunction_] = uType;
97  //std::cout << nJunction_ << " " << i << " "
98  // << lType << " " << uType << std::endl;
99  ++nJunction_;
100  }
101  }
102 
103  read<int>(in, "hasAutoCorr", hasAutoCorr_);
104 
105  // Read autocorrelation parameters
106  if (hasAutoCorr_) {
107  read<int>(in, "autoCorrCapacity", autoCorrCapacity_);
108  read<std::string>(in, "autoCorrName", outputFileName_);
109 
110  // Allocate memory for autocorrelation function of reptation steps
111  int moleculeCapacity = simulation().species(speciesId_).capacity();
112  acceptedStepsAccumulators_.allocate(moleculeCapacity);
113  for (int i = 0; i < moleculeCapacity; i++)
114  {
115  acceptedStepsAccumulators_[i].setParam(autoCorrCapacity_);
116  }
117  }

187  /*
188  * Generate, attempt and accept or reject a Monte Carlo move.
189  */
190  bool CfbReptationMove::move()
191  {
192  Vector oldPos, newPos;
193  double rosen_r, rosen_f, ratio;
194  double energy_r, energy_f;
195  Atom *tailPtr; // pointer to the tail atom (to be removed)
196  Atom *atomPtr; // resetable atom pointer
197  Molecule *molPtr; // pointer to randomly chosen molecule
198  int length, sign, headId, tailId, oldType, i;
199  bool accept;
200 
201  incrementNAttempt();
202 
203  // Choose a molecule at random
204  molPtr = &(system().randomMolecule(speciesId_));
205  length = molPtr->nAtom();
206 
207  // Choose which chain end to regrow
208  if (random().uniform(0.0, 1.0) > 0.5) {
209  sign = +1;
210  headId = length - 1;
211  tailId = 0;
212  } else {
213  sign = -1;
214  headId = 0;
215  tailId = length - 1;
216  }
217 
218  // Store current position and type of tail atom
219  tailPtr = &(molPtr->atom(tailId));
220  oldPos = tailPtr->position();
221  oldType = tailPtr->typeId();
222 
223  // Delete tail atom
224  atomPtr = tailPtr + sign;
225  deleteEndAtom(tailPtr, atomPtr, bondTypeId_, rosen_r, energy_r);
226  #ifndef INTER_NOPAIR
227  // Delete from McSystem cell list
228  system().pairPotential().deleteAtom(*tailPtr);
229  #endif
230 
231  // Regrow head, using tailPtr to store properties of the new head.
232  atomPtr = &(molPtr->atom(headId));
233  tailPtr->setTypeId(atomPtr->typeId());
234  tailPtr->mask().clear();
235  if (maskPolicy_ == MaskBonded) {
236  tailPtr->mask().append(*atomPtr);
237  }
238  addEndAtom(tailPtr, atomPtr, bondTypeId_, rosen_f, energy_f);
239 
240  // Calculate junction factor for heteropolymers.
241  double jFactor = 1.0;
242  if (nJunction_ > 0) {
243  jFactor = junctionFactor(molPtr, sign);
244  // Upon return, type ids are restored to original values.
245  }
246 
247  // Decide whether to accept or reject
248  ratio = jFactor * rosen_f / rosen_r;
249  accept = random().metropolis(ratio);
250 
251  // Restore original type and connectivity mask for tail Atom
252  tailPtr->setTypeId(oldType);
253  tailPtr->mask().clear();
254  if (maskPolicy_ == MaskBonded) {
255  atomPtr = tailPtr + sign;
256  tailPtr->mask().append(*atomPtr);
257  }
258 
259  if (accept) {
260 
261  // Store new head position
262  newPos = tailPtr->position();
263 
264  // Shift position of tail
265  atomPtr = tailPtr + sign;
266  tailPtr->position() = atomPtr->position();
267 
268  #ifndef INTER_NOPAIR
269  // Add tail back to system cell list
270  system().pairPotential().addAtom(*tailPtr);
271  #endif
272 
273  // Shift atom positions towards the head
274  for (i=1; i < length - 1; ++i) {
275  //system().moveAtom(*atomPtr, (atomPtr+sign)->position());
276  atomPtr->position() = (atomPtr+sign)->position();
277  #ifndef INTER_NOPAIR
278  system().pairPotential().updateAtomCell(*atomPtr);
279  #endif
280  atomPtr += sign;
281  }
282 
283  // Move head atom to new chosen position
284  // system().moveAtom(*atomPtr, newPos);
285  atomPtr->position() = newPos;
286  #ifndef INTER_NOPAIR
287  system().pairPotential().updateAtomCell(*atomPtr);
288  #endif
289 
290  // Increment the number of accepted moves.
291  incrementNAccept();
292 
293  if (hasAutoCorr_)
294  {
295  // Store step direction in autocorrelator
296  acceptedStepsAccumulators_[molPtr->id()].sample((double) sign);
297  }
298  } else {
299 
300  // Restore old position of tail
301  tailPtr->position() = oldPos;
302 
303  #ifndef INTER_NOPAIR
304  // Add tail back to System cell list.
305  system().pairPotential().addAtom(*tailPtr);
306  #endif
307 
308  }
309 
310  return accept;
311 
312  }

317  double CfbReptationMove::junctionFactor(Molecule* molPtr, int sign)
318  {
319 
320  double oldEnergy, newEnergy, factor;
321  Atom* hAtomPtr; // Pointer to atom nearer head
322  int hType; // type id of atom nearer head
323  int tType; // type id of atom nearer tail
324  int i; // junction index
325  int j; // id of atom at below junction (lower)
326 
327 
328  // Calculate factor by looping over junctions
329  factor = 1.0;
330  for (i = 0; i < nJunction_; ++i) {
331  j = junctions_[i];
332  if (sign == 1) {
333  hAtomPtr = &(molPtr->atom(j+1));
334  tType = lTypes_[i];
335  } else {
336  hAtomPtr = &(molPtr->atom(j));
337  tType = uTypes_[i];
338  }
339 
340  #ifndef INTER_NOPAIR
341  oldEnergy = system().pairPotential().atomEnergy(*hAtomPtr);
342  #endif
343 
344  #ifdef INTER_EXTERNAL
345  oldEnergy += system().externalPotential().atomEnergy(*hAtomPtr);
346  #endif
347 
348  #ifndef INTER_NOPAIR
349  hAtomPtr->setTypeId(tType);
350  newEnergy = system().pairPotential().atomEnergy(*hAtomPtr);
351  #endif
352 
353  #ifdef INTER_EXTERNAL
354  newEnergy += system().externalPotential().atomEnergy(*hAtomPtr);
355  #endif
356 
357  #ifndef INTER_NOPAIR
358  factor *= boltzmann(newEnergy - oldEnergy);
359  #endif
360  }
361 
362  // Revert modified atom type Ids to original values
363  #ifndef INTER_NOPAIR
364  for (i = 0; i < nJunction_; ++i) {
365  j = junctions_[i];
366  if (sign == 1) {
367  hAtomPtr = &(molPtr->atom(j+1));
368  hType = uTypes_[i];
369  } else {
370  hAtomPtr = &(molPtr->atom(j));
371  hType = lTypes_[i];
372  }
373  hAtomPtr->setTypeId(hType);
374  }
375  return factor;
376 
377  #else //ifdef INTER_NOPAIR
378  return 1.0;
379  #endif
380  }

385  void CfbReptationMove::output()
386  {
387  if (hasAutoCorr_)
388  {
389  DArray< double > autoCorrAvg;
390  autoCorrAvg.allocate(autoCorrCapacity_);
391 
392  // Calculate average autocorrelation function over all
393  // accumulators
394  for (int i = 0; i < autoCorrCapacity_; i++) {
395  int nAvg = 0;
396  autoCorrAvg[i] = 0;
397  for (int j = 0; j < system().nMolecule(speciesId_); j++)
398  {
399  if (acceptedStepsAccumulators_[j].nSample() > i)
400  {
401  nAvg++;
402  autoCorrAvg[i] += acceptedStepsAccumulators_[j].
403  autoCorrelation(i);
404  }
405  }
406  autoCorrAvg[i] /= nAvg;
407  }
408 
409  std::ofstream outputFile;
410 
411  // Write out average autocorrelation
412  system().fileMaster().openOutputFile(outputFileName_+".dat",
413  outputFile);
414 
415  for (int i = 0; i < autoCorrCapacity_; i++)
416  {
417  outputFile << Int(i);
418  write<double>(outputFile, autoCorrAvg[i]);
419  outputFile << std::endl;
420  }
421 
422  outputFile.close();
423 
424  // Sum over autocorrelation (only positive time lags)
425  // Count first element with a factor 1/2, due to symmetry
426  double acSum = 0;
427  acSum = 0.5*autoCorrAvg[0];
428  for (int i = 1; i < autoCorrCapacity_; i++)
429  acSum += autoCorrAvg[i];
430 
431  // write sum in .ave file
432  system().fileMaster().openOutputFile(outputFileName_+".ave",
433  outputFile);
434  outputFile << "autoCorrSum " << acSum << std::endl;
435  outputFile.close();
436  }
437  }
