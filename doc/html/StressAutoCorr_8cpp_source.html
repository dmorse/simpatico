109  /*
110  * Sample the stress tensor.
111  */
112  void StressAutoCorr::sample(long iStep)
113  {
114  if (isAtInterval(iStep)) {
115  Simulation& sim = simulation();
116  sim.computeVirialStress();
117  sim.computeKineticStress();
118 
119  if (sim.domain().isMaster()) {
120  if (!accumulatorPtr_) {
121  UTIL_THROW("Null accumulatorPtr_ on master");
122  }
123 
124  Tensor virial = sim.virialStress();
125  Tensor kinetic = sim.kineticStress();
126  Tensor total = total.add(virial, kinetic);
127 
128  // Remove trace
129  double pressure = 0.0;
130  int i, j;
131  for (i = 0; i < Dimension; ++i) {
132  pressure += total(i,i);
133  }
134  pressure = pressure/double(Dimension);
135  for (i = 0; i < Dimension; ++i) {
136  total(i,i) -= pressure;
137  }
138 
139  double factor = sqrt(sim.boundary().volume()/10.0);
140  for (i = 0; i < Dimension; ++i) {
141  for (j = 0; j < Dimension; ++j) {
142  total(i,j) *= factor;
143  }
144  }
145 
146  accumulatorPtr_->sample(total);
147  }
148  }
149  }
