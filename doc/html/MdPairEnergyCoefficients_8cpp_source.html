155  * Evaluate contributions to accumulators.
156  */
157  void MdPairEnergyCoefficients::sample(long iStep)
158  {
159  if (isAtInterval(iStep)) {
160 
161  #ifndef INTER_NOPAIR
162 // if (!system().isPairListCurrent()) {
163 // system().buildPairList();
164 // }
165 
166  // First clear neighbor list
167  clear();
168 
169  // Loop over pairs and construct neighbor list per molecule
170  PairIterator iter;
171  Atom *atom0Ptr;
172  Atom *atom1Ptr;
173 
174  for (pairListPtr_->begin(iter); iter.notEnd(); ++iter) {
175  iter.getPair(atom0Ptr, atom1Ptr);
176  if (selector_.match(*atom0Ptr, *atom1Ptr)) {
177  Pair< Atom * > atomPair;
178  Molecule *moleculePtr;
179  Species *speciesPtr;
180  int speciesId, moleculeId;
181 
182  atomPair[0] = atom0Ptr;
183  atomPair[1] = atom1Ptr;
184 
185  // Store pair in neighbor list of boths atoms' molecules
186  // (do double counting, since pair list counts only unique
187  // pairs)
188  moleculePtr = & atom0Ptr->molecule();
189  speciesPtr = & moleculePtr->species();
190  speciesId = speciesPtr->id();
191  moleculeId = moleculePtr->id();
192 
193  moleculeNeighbors_[speciesId][moleculeId].
194  append(atomPair);
195 
196  moleculePtr = & atom1Ptr->molecule();
197  speciesPtr = & moleculePtr->species();
198  speciesId = speciesPtr->id();
199  moleculeId = moleculePtr->id();
200 
201  // store atomPair in reverse order
202  atomPair[0] = atom1Ptr;
203  atomPair[1] = atom0Ptr;
204 
205  moleculeNeighbors_[speciesId][moleculeId].
206  append(atomPair);
207  }
208  } // end pair loop
209 
210  // Now loop over all molecules
211  double moleculePESq = 0.0;
212  double pairEnergy=0.0;
213  double twoMoleculePESq=0.0;
214 
215  double rsq;
216  int iSpecies1, iMolecule1;
217 
218  for (iSpecies1 = 0; iSpecies1 < nSpecies_; ++iSpecies1) {
219  Species *species1Ptr;
220 
221  species1Ptr = &system().simulation().species(iSpecies1);
222  for (iMolecule1 = 0; iMolecule1 < species1Ptr->capacity();
223  ++iMolecule1) {
224  int iSpecies2, iMolecule2;
225 
226  double moleculePairEnergy=0.0;
227 
228  // clear array for pair energy with neighboring molecules
229  for (iSpecies2 = 0; iSpecies2 < nSpecies_; ++iSpecies2) {
230  Species *species2Ptr;
231 
232  species2Ptr = &system().simulation().species(iSpecies2);
233  for (iMolecule2 = 0; iMolecule2 < species2Ptr->capacity();
234  ++iMolecule2) {
235  twoMoleculePairEnergy_[iSpecies2][iMolecule2]=0.0;
236  }
237  }
238 
239  // Loop over this molecule's neighbors
240  DSArray< Pair< Atom *> > *neighborsPtr;
241  ConstArrayIterator< Pair< Atom *> > it;
242 
243  neighborsPtr=&moleculeNeighbors_[iSpecies1][iMolecule1];
244  for (neighborsPtr->begin(it); it.notEnd(); ++it) {
245  Atom *atom0Ptr, *atom1Ptr;
246  atom0Ptr = (*it)[0];
247  atom1Ptr = (*it)[1];
248 
249  if (selector_.match(*atom0Ptr,*atom1Ptr)) {
250  double energy;
251  Species *species2Ptr;
252  Molecule *molecule2Ptr;
253  int species2Id,molecule2Id;
254 
255  // Load second atom's molecule and species
256  molecule2Ptr = &atom1Ptr->molecule();
257  species2Ptr = &atom1Ptr->molecule().species();
258  species2Id = species2Ptr->id();
259  molecule2Id = molecule2Ptr->id();
260 
261  rsq = boundaryPtr_->distanceSq(atom0Ptr->position(),
262  atom1Ptr->position());
263  energy = pairPotentialPtr_->energy(rsq,
264  atom0Ptr->typeId(), atom1Ptr->typeId());
265 
266  // increment pair energy of this molecule pair
267  twoMoleculePairEnergy_[species2Id][molecule2Id] +=
268  energy;
269  }
270  } // end neighbors loop
271 
272  // sum the molecular pair energies and their squares
273  for (iSpecies2 = 0; iSpecies2 < nSpecies_; ++iSpecies2) {
274  Species *species2Ptr;
275 
276  species2Ptr = &system().simulation().species(iSpecies2);
277  for (iMolecule2 = 0; iMolecule2 < species2Ptr->capacity();
278  ++iMolecule2) {
279  double energy =
280  twoMoleculePairEnergy_[iSpecies2][iMolecule2];
281  moleculePairEnergy += energy;
282  twoMoleculePESq += energy*energy;
283  }
284 
285  }
286 
287  // accumulate total molecular pair energy and its square
288  pairEnergy += moleculePairEnergy;
289  moleculePESq += moleculePairEnergy*moleculePairEnergy;
290 
291  } // end molecule loop
292 
293  } // end species loop
294 
295  // Sample sum of pair energies
296  pairEnergyAccumulator_.sample(pairEnergy);
297 
298  // Sample sum of squares of molecular pair energy
299  moleculePESqAccumulator_.sample(moleculePESq);
300 
301  // Sample sum of squares of two molecule pair energy
302  twoMoleculePESqAccumulator_.sample(twoMoleculePESq);
303 
304  // Sample square of total pair energy
305  pESqAccumulator_.sample(pairEnergy*pairEnergy);
306 
307  outputFile_ << Dbl(pairEnergy);
308  outputFile_ << Dbl(moleculePESq);
309  outputFile_ << Dbl(twoMoleculePESq);
310  outputFile_ << Dbl(pairEnergy*pairEnergy);
311 
312  #endif
313  outputFile_ << std::endl;
314  }
