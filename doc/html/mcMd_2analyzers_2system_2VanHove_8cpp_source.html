124  void VanHove::sample(long iStep)
125  {

128  Vector position;
129  std::complex<double> expFactor;
130  double product, coeff;
131  System::ConstMoleculeIterator molIter;
132  Molecule::ConstAtomIterator atomIter;
133  int nSpecies, iSpecies, i;
134 
135  makeWaveVectors();
136 
137  // Set all Fourier modes to zero
138  for (i = 0; i < nWave_; ++i) {
139  fourierModes_[i] = std::complex<double>(0.0, 0.0);
140  }
141 
142  // Loop over all atoms to calculate Fourier modes
143  nSpecies = system().simulation().nSpecies();
144  for (iSpecies = 0; iSpecies < nSpecies; ++iSpecies) {
145  system().begin(iSpecies, molIter);
146  for ( ; molIter.notEnd(); ++molIter) {
147  molIter->begin(atomIter);
148  for ( ; atomIter.notEnd(); ++atomIter) {
149  position = atomIter->position();
150  coeff = atomTypeCoeffs_[atomIter->typeId()];
151 
152  // Loop over wavevectors
153  for (i = 0; i < nWave_; ++i) {
154 
155  product = position.dot(waveVectors_[i]);
156  expFactor = exp( product*Constants::Im );
157  fourierModes_[i] += (coeff*expFactor);
158 
159  }
160 
161  }
162  }
163 
164  }
165 
166  // Add Fourier modes to autocorrelation accumulators
167  double sqrtV = sqrt(system().boundary().volume());
168  for (i = 0; i < nWave_; ++i) {
169  accumulators_[i].sample(fourierModes_[i]/sqrtV);
170  }
171 
172  }
