168  template < class Interaction >
169  double McExternalPerturbation<Interaction>::derivative(int i) const
170  {
171  // Preconditions
172  if (i > nParameters_) {
173  UTIL_THROW("perturbation parameter index is out of bounds");
174  }
175 
176  if (fabs(parameter_[i] - parameter(i)) > 1.0E-8) {
177  UTIL_THROW("Perturbation parameter is not set correctly");
178  }
179 
180  double energy;
181  System::MoleculeIterator molIter;
182  Molecule::AtomIterator atomIter;
183 
184  energy = 0.0;
185 
186  for (int iSpec=0; iSpec < system().simulation().nSpecies(); ++iSpec){
187  for (system().begin(iSpec, molIter); molIter.notEnd(); ++molIter){
188  for (molIter->begin(atomIter); atomIter.notEnd(); ++atomIter) {
189  energy += interaction().energy(atomIter->position(), atomIter->typeId());
190 
191  }
192  }
193  }
194 
195  // Multiply by the temperature factor.
196  if (system().energyEnsemble().isIsothermal()) {
197  energy *= system().energyEnsemble().beta();
198  } else {
199  UTIL_THROW("Non isothermal ensemble.");
200  }
201 
202  return energy/parameter_[i];
