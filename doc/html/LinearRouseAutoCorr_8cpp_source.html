141  /*
142  * Evaluate end-to-end vectors of all chains, add to ensemble.
143  */
144  void LinearRouseAutoCorr::setup()
145  {
146 
147  if (!isInitialized_) {
148  UTIL_THROW("Object is not intitialized");
149  }
150 
151  // Get number of molecules and number of atoms per molecule.
152  nMolecule_ = system().nMolecule(speciesId_);
153  if (nAtom_ <= 1) UTIL_THROW("Number of atoms per molecule < 2.");
154 
155  // Initialize mode projection coefficients.
156  double qMode = acos(-1.0) * double(p_) / double(nAtom_ - 1);
157  if (p_ == 0) {
158  for (int j = 0; j < nAtom_; ++j)
159  projector_[j] = 1.0 / double(nAtom_);
160  } else {
161  for (int j = 0; j < nAtom_; ++j)
162  projector_[j] = 1.0 / double(nAtom_) * cos(qMode*double(j));
163  }
164  projector_[0] -= 0.5/double(nAtom_);
165  projector_[nAtom_ - 1] -= 0.5 / double(nAtom_);
166 
167  // Initialize the AutoCorrArray object
168  accumulator_.setNEnsemble(nMolecule_);
169 
170  accumulator_.clear();
171  }

173  /*
174  * Evaluate end-to-end vectors of all chains, add to ensemble.
175  */
176  void LinearRouseAutoCorr::sample(long iStep)
177  {
178  if (isAtInterval(iStep)) {
179 
180  Molecule* moleculePtr;
181  Vector r1, r2, dR, atomPos;
182  int i, j;
183 
184  // Confirm that nMolecule has remained constant
185  if (nMolecule_ != system().nMolecule(speciesId_)) {
186  UTIL_THROW("Number of molecules has changed.");
187  }
188 
189  // Loop over molecules
190  for (i = 0; i < nMolecule_; ++i) {
191  moleculePtr = &(system().molecule(speciesId_, i));
192 
193  // Retrace non-periodic shape and calculate coefficients
194  atomPos = moleculePtr->atom(0).position();
195  dR.multiply(atomPos, projector_[0]);
196  data_[i] = dR;
197  for (j = 1; j < nAtom_; j++) {
198  r1 = moleculePtr->atom(j-1).position();
199  r2 = moleculePtr->atom(j).position();
200  system().boundary().distanceSq(r2, r1, dR);
201  atomPos += dR;
202  dR.multiply(atomPos, projector_[j]);
203  data_[i] += dR;
204  }
205  }
206 
207  accumulator_.sample(data_);
208 
209  } // if isAtInterval
210
