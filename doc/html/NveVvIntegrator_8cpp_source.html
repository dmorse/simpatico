83  /*
84  * Verlet MD NVE integrator step
85  *
86  * This method implements the algorithm:
87  *
88  * vm(n) = v(n) + 0.5*a(n)*dt
89  * x(n+1) = x(n) + vm(n)*dt
90  *
91  * calculate acceleration a(n+1)
92  *
93  * v(n+1) = vm(n) + 0.5*a(n+1)*dt
94  *
95  * where x is position, v is velocity, and a is acceleration.
96  */
97  void NveVvIntegrator::step()
98  {
99  Vector dv;
100  Vector dr;
101  System::MoleculeIterator molIter;
102  double prefactor;
103  //double dtHalf = 0.5*dt_;
104  int iSpecies, nSpecies;
105 
106  nSpecies = simulation().nSpecies();
107 
108  // 1st half velocity Verlet, loop over atoms
109  #if USE_ITERATOR
110  Molecule::AtomIterator atomIter;
111  for (iSpecies=0; iSpecies < nSpecies; ++iSpecies) {
112  system().begin(iSpecies, molIter);
113  for ( ; molIter.notEnd(); ++molIter) {
114  for (molIter->begin(atomIter); atomIter.notEnd(); ++atomIter) {
115 
116  prefactor = prefactors_[atomIter->typeId()];
117 
118  dv.multiply(atomIter->force(), prefactor);
119  atomIter->velocity() += dv;
120 
121  dr.multiply(atomIter->velocity(), dt_);
122  atomIter->position() += dr;
123 
124  }
125  }
126  }
127  #else
128  Atom* atomPtr;
129  int ia;
130  for (iSpecies=0; iSpecies < nSpecies; ++iSpecies) {
131  system().begin(iSpecies, molIter);
132  for ( ; molIter.notEnd(); ++molIter) {
133  for (ia=0; ia < molIter->nAtom(); ++ia) {
134  atomPtr = &molIter->atom(ia);
135  prefactor = prefactors_[atomPtr->typeId()];
136 
137  dv.multiply(atomPtr->force(), prefactor);
138  atomPtr->velocity() += dv;
139  dr.multiply(atomPtr->velocity(), dt_);
140 
141  atomPtr->position() += dr;
142 
143  }
144  }
145  }
146  #endif
147 
148  system().calculateForces();
149 
150  // 2nd half velocity Verlet, loop over atoms
151  #if USE_ITERATOR
152  for (iSpecies=0; iSpecies < nSpecies; ++iSpecies) {
153  system().begin(iSpecies, molIter);
154  for ( ; molIter.notEnd(); ++molIter) {
155  for (molIter->begin(atomIter); atomIter.notEnd(); ++atomIter) {
156  prefactor = prefactors_[atomIter->typeId()];
157  dv.multiply(atomIter->force(), prefactor);
158  atomIter->velocity() += dv;
159  }
160  }
161  }
162  #else
163  for (iSpecies=0; iSpecies < nSpecies; ++iSpecies) {
164  system().begin(iSpecies, molIter);
165  for ( ; molIter.notEnd(); ++molIter)
166  {
167  for (ia=0; ia < molIter->nAtom(); ++ia) {
168  atomPtr = &molIter->atom(ia);
169  prefactor = prefactors_[atomPtr->typeId()];
170  dv.multiply(atomPtr->force(), prefactor);
171  atomPtr->velocity() += dv;
172  }
173  }
174  }
175  #endif
176 
177  #ifndef INTER_NOPAIR
178  if (!system().pairPotential().isPairListCurrent()) {
179  system().pairPotential().buildPairList();
180  }
181  #endif

183  }
