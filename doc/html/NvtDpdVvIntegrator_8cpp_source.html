172  /*
173  * Compute new dissipative and (optionally) random forces.
174  *
175  * Dissipative and random forces are stored in separate arrays.
176  * If computeRandom == true, recompute random as well as dissipative
177  * forces.
178  */
179  void NvtDpdVvIntegrator::computeDpdForces(bool computeRandom)
180  {
181  Vector e; // unit vector (r0 - r1)/|r0 - r1|;
182  Vector f; // force vector
183  Vector dv; // difference in velocities v0 - v1;
184  double rsq; // square of distance between atoms.
185  double r; // distance between atoms.
186  double fr; // magnitude of random pair force.
187  double fd; // magnitude of dissipative pair force.
188  double wr; // weighting function for random forces.
189  Atom* atom0Ptr;
190  Atom* atom1Ptr;
191  PairIterator iter;
192  int i;
193 
194  // Set all dissipative forces to zero.
195  for (i=0; i < atomCapacity_; ++i) {
196  dissipativeForces_[i].zero();
197  }
198  if (computeRandom) {
199  for (i=0; i < atomCapacity_; ++i) {
200  randomForces_[i].zero();
201  }
202  }
203 
204  // Iterator over atom pairs
205  for (pairListPtr_->begin(iter); iter.notEnd(); ++iter) {
206  iter.getPair(atom0Ptr, atom1Ptr);
207  rsq = boundaryPtr_->distanceSq(atom0Ptr->position(),
208  atom1Ptr->position(), e);
209  if (rsq < cutoffSq_) {
210  r = sqrt(rsq);
211  e /= r;
212 
213  #if MCMD_DPD_TYPE == 0
214  wr = 1.0;
215  #endif
216  #if MCMD_DPD_TYPE == 1
217  wr = 1.0 - (r/cutoff_);
218  #endif
219  #if MCMD_DPD_TYPE == 2
220  wr = 1.0 - rsq/cutoffSq_;
221  #endif
222 
223  // Add random forces to atomic vectors.
224  if (computeRandom) {
225  fr = sigma_*wr*randomPtr_->gaussian();
226  f.multiply(e, fr);
227  randomForces_[atom0Ptr->id()] += f;
228  randomForces_[atom1Ptr->id()] -= f;
229  }
230 
231  // Add dissipative forces to array
232  dv.subtract(atom0Ptr->velocity(), atom1Ptr->velocity());
233  fd = -gamma_*wr*wr*dv.dot(e);
234  f.multiply(e, fd);
235  dissipativeForces_[atom0Ptr->id()] += f;
236  dissipativeForces_[atom1Ptr->id()] -= f;
237 
238  }
239 
240  }

245  * DPD integrator time step.
246  */
247  void NvtDpdVvIntegrator::step()
248  {
249  Vector dv;
250  Vector dr;
251  Vector f;
252  System::MoleculeIterator molIter;
253  int iSpecies, nSpecies, atomId;
254 
255  nSpecies = simulation().nSpecies();
256 
257  // 1st half velocity Verlet, loop over atoms
258  Molecule::AtomIterator atomIter;
259  for (iSpecies=0; iSpecies < nSpecies; ++iSpecies) {
260  for (system().begin(iSpecies, molIter); molIter.notEnd(); ++molIter)
261  {
262  for (molIter->begin(atomIter); atomIter.notEnd(); ++atomIter) {
263  atomId = atomIter->id();
264 
265  // f = (conservative atomic) + random + dissipative force
266  f.add(atomIter->force(), dissipativeForces_[atomId]);
267  f += randomForces_[atomId];
268 
269  // Velocity increment dv = force*0.5*dt/mass
270  dv.multiply(f, dtMinvFactors_[atomIter->typeId()]);
271  atomIter->velocity() += dv;
272 
273  dr.multiply(atomIter->velocity(), dt_);
274  atomIter->position() += dr;
275 
276  }
277  }
278  }
279 
280  #if 0
281  #ifndef INTER_NOPAIR
282  if (!system().pairPotential().isPairListCurrent()) {
283  system().pairPotential().buildPairList();
284  }
285  #endif
286  #endif
287 
288  // Note: MdSystem().calculateForces() rebuilds the pair list as needed.
289 
290  // Calculate all new forces
291  system().calculateForces();
292  computeDpdForces(true);
293 
294  // 2nd half velocity Verlet, loop over atoms
295  for (iSpecies=0; iSpecies < nSpecies; ++iSpecies) {
296  system().begin(iSpecies, molIter);
297  for ( ; molIter.notEnd(); ++molIter) {
298  for (molIter->begin(atomIter); atomIter.notEnd(); ++atomIter) {
299  atomId = atomIter->id();
300 
301  // f = conservative + random + dissipative force
302  f.add(atomIter->force(), dissipativeForces_[atomId]);
303  f += randomForces_[atomId];
304 
305  // Increment velocity by dv = f*0.5*dt/mass
306  dv.multiply(f, dtMinvFactors_[atomIter->typeId()]);
307  atomIter->velocity() += dv;
308 
309  }
310  }
311  }
312 
313  // Recompute disipative force (but not random forces).
314  computeDpdForces(false);
315 
316  }
