45  void CfbReptateMove::readParameters(std::istream& in)
46  {
47  // Read parameters
48  readProbability(in);
49  CfbLinear::readParameters(in); // read speciesId_ and nTrial_
50 
51  hasAutoCorr_ = false; // default value
52  readOptional<bool>(in, "hasAutoCorr", hasAutoCorr_);
53  if (hasAutoCorr_) {
54  read<int>(in, "autoCorrCapacity", autoCorrCapacity_);
55  read<std::string>(in, "outputFileName", outputFileName_);
56  }
57 
58  // Identify bond type, check that it is same for all bonds
59  Species* speciesPtr = &simulation().species(speciesId());
60  int nBond = speciesPtr->nBond();
61  bondTypeId_ = speciesPtr->speciesBond(0).typeId();
62  for (int i = 1; i < nBond; ++i) {
63  if (bondTypeId_ != speciesPtr->speciesBond(i).typeId()) {
64  UTIL_THROW("Unequal bond type ids");
65  }
66  }
67 
68  // Identify policy for masking nonbonded interactions.
69  maskPolicy_ = simulation().maskedPairPolicy();
70 
71  // Allocate memory for junction arrays
72  int nAtom = speciesPtr->nAtom();
73  junctions_.allocate(nAtom-1);
74  lTypes_.allocate(nAtom-1);
75  uTypes_.allocate(nAtom-1);
76 
77  // Identify junctions between atoms of different types
78  int lType, uType;
79  nJunction_ = 0;
80  for (int i = 0; i < nAtom - 1; ++i) {
81  lType = speciesPtr->atomTypeId(i);
82  uType = speciesPtr->atomTypeId(i+1);
83  if (lType != uType) {
84  junctions_[nJunction_] = i;
85  lTypes_[nJunction_] = lType;
86  uTypes_[nJunction_] = uType;
87  ++nJunction_;
88  }
89  }
90 
91  if (hasAutoCorr_) {
92  // Allocate memory for autocorrelation accumulators
93  int moleculeCapacity = simulation().species(speciesId()).capacity();
94  accumulators_.allocate(moleculeCapacity);
95  for (int i = 0; i < moleculeCapacity; i++) {
96  accumulators_[i].setParam(autoCorrCapacity_);
97  }
98  }

159  /*
160  * Generate, attempt and accept or reject a Monte Carlo move.
161  */
162  bool CfbReptateMove::move()
163  {
164  Vector oldPos, newPos;
165  double rosen_r, rosen_f, ratio;
166  double energy_r, energy_f;
167  Atom *tailPtr; // pointer to the tail atom (to be removed)
168  Atom *atomPtr; // resetable atom pointer
169  Molecule *molPtr; // pointer to chosen molecule
170  int nAtom, sign, headId, tailId, oldType, i;
171  bool accept;
172 
173  incrementNAttempt();
174 
175  // Choose a molecule of specific species at random
176  molPtr = &(system().randomMolecule(speciesId()));
177  nAtom = molPtr->nAtom();
178 
179  // Choose which chain end to regrow.
180  // "tail" = deletion end, "head" = addition end
181  if (random().uniform(0.0, 1.0) > 0.5) {
182  sign = +1;
183  headId = nAtom - 1;
184  tailId = 0;
185  } else {
186  sign = -1;
187  headId = 0;
188  tailId = nAtom - 1;
189  }
190 
191  // Store current position and type of tail atom
192  tailPtr = &(molPtr->atom(tailId));
193  oldPos = tailPtr->position();
194  oldType = tailPtr->typeId();
195 
196  // Delete tail atom
197  deleteAtom(*molPtr, tailId, -sign, rosen_r, energy_r);
198  #ifndef INTER_NOPAIR
199  // Delete from McSystem cell list
200  system().pairPotential().deleteAtom(*tailPtr);
201  #endif
202 
203  // Regrow head, using tailPtr to store properties of the new head.
204  atomPtr = &(molPtr->atom(headId)); // new pivot atom
205  tailPtr->setTypeId(atomPtr->typeId()); // new head atom
206  tailPtr->mask().clear();
207  if (maskPolicy_ == MaskBonded) {
208  tailPtr->mask().append(*atomPtr);
209  }
210  addAtom(*molPtr, *tailPtr, *atomPtr, headId, sign, rosen_f, energy_f);
211 
212  // Calculate junction factor for heteropolymers.
213  double jFactor;
214  if (nJunction_ > 0) {
215  jFactor = junctionFactor(molPtr, sign);
216  // Upon return, type ids are restored to original values.
217  } else {
218  jFactor = 1.0;
219  }
220 
221  // Decide whether to accept or reject
222  ratio = jFactor * rosen_f / rosen_r;
223  accept = random().metropolis(ratio);
224 
225  // Restore original type and connectivity mask for tail Atom
226  tailPtr->setTypeId(oldType);
227  tailPtr->mask().clear();
228  if (maskPolicy_ == MaskBonded) {
229  atomPtr = tailPtr + sign;
230  tailPtr->mask().append(*atomPtr);
231  }
232 
233  if (accept) {
234 
235  // Store position of new head atom
236  newPos = tailPtr->position();
237 
238  // Shift position of tail
239  atomPtr = tailPtr + sign;
240  tailPtr->position() = atomPtr->position();
241  #ifndef INTER_NOPAIR
242  // Add tail back to system cell list
243  system().pairPotential().addAtom(*tailPtr);
244  #endif
245 
246  // Shift atom positions towards the head
247  for (i = 1; i < nAtom - 1; ++i) {
248  atomPtr->position() = (atomPtr+sign)->position();
249  #ifndef INTER_NOPAIR
250  system().pairPotential().updateAtomCell(*atomPtr);
251  #endif
252  atomPtr += sign;
253  }
254  assert(atomPtr == &molPtr->atom(headId));
255 
256  // Move head atom to new chosen position
257  atomPtr->position() = newPos;
258  #ifndef INTER_NOPAIR
259  system().pairPotential().updateAtomCell(*atomPtr);
260  #endif
261 
262  // Increment the number of accepted moves.
263  incrementNAccept();
264 
265  if (hasAutoCorr_) {
266  int molId = molPtr->id();
267  double rsign = (double) sign;
268  accumulators_[molId].sample(rsign);
269  //accumulators_[molPtr->id()].sample((double) sign);
270  }
271 
272  } else {
273 
274  // Restore old position of tail
275  tailPtr->position() = oldPos;
276  #ifndef INTER_NOPAIR
277  // Add tail back to System cell list.
278  system().pairPotential().addAtom(*tailPtr);
279  #endif
280 
281  }
282 
283  return accept;
284  }

289  double CfbReptateMove::junctionFactor(Molecule* molPtr, int sign)
290  {
291  double oldEnergy, newEnergy, factor;
292  Atom* hAtomPtr; // Pointer to atom nearer head
293  int hType; // type id of atom nearer head
294  int tType; // type id of atom nearer tail
295  int i; // junction index
296  int j; // id of atom at below junction (lower)
297 
298  // Calculate factor by looping over junctions
299  factor = 1.0;
300  for (i = 0; i < nJunction_; ++i) {
301  j = junctions_[i];
302  if (sign == 1) {
303  hAtomPtr = &(molPtr->atom(j+1));
304  tType = lTypes_[i];
305  } else {
306  hAtomPtr = &(molPtr->atom(j));
307  tType = uTypes_[i];
308  }
309 
310  #ifndef INTER_NOPAIR
311  oldEnergy = system().pairPotential().atomEnergy(*hAtomPtr);
312  #else
313  oldEnergy = 0.0;
314  #endif
315  #ifdef INTER_EXTERNAL
316  oldEnergy += system().externalPotential().atomEnergy(*hAtomPtr);
317  #endif
318 
319  #ifndef INTER_NOPAIR
320  hAtomPtr->setTypeId(tType);
321  newEnergy = system().pairPotential().atomEnergy(*hAtomPtr);
322  #else
323  newEnergy = 0.0;
324  #endif
325  #ifdef INTER_EXTERNAL
326  newEnergy += system().externalPotential().atomEnergy(*hAtomPtr);
327  #endif
328 
329  factor *= boltzmann(newEnergy - oldEnergy);
330  }
331 
332  // Revert modified atom type Ids to original values
333  #ifndef INTER_NOPAIR
334  for (i = 0; i < nJunction_; ++i) {
335  j = junctions_[i];
336  if (sign == 1) {
337  hAtomPtr = &(molPtr->atom(j+1));
338  hType = uTypes_[i];
339  } else {
340  hAtomPtr = &(molPtr->atom(j));
341  hType = lTypes_[i];
342  }
343  hAtomPtr->setTypeId(hType);
344  }
345  return factor;
346 
347  #else //ifdef INTER_NOPAIR
348  return 1.0;
349  #endif
350  }

355  void CfbReptateMove::output()
356  {
357  if (hasAutoCorr_) {
358  DArray< double > autoCorrAvg;
359  autoCorrAvg.allocate(autoCorrCapacity_);
360 
361  // Compute average autocorrelation function over molecules
362  for (int i = 0; i < autoCorrCapacity_; i++) {
363  int nAvg = 0;
364  autoCorrAvg[i] = 0;
365  for (int j = 0; j < system().nMolecule(speciesId()); j++) {
366  if (accumulators_[j].nSample() > i) {
367  nAvg++;
368  autoCorrAvg[i] += accumulators_[j].autoCorrelation(i);
369  }
370  }
371  autoCorrAvg[i] /= nAvg;
372  }
373 
374  // Write out average autocorrelation
375  std::ofstream outputFile;
376  std::string fileName = outputFileName_;
377  fileName += ".dat";
378  system().fileMaster().openOutputFile(fileName, outputFile);
379  for (int i = 0; i < autoCorrCapacity_; i++) {
380  outputFile << Int(i);
381  write<double>(outputFile, autoCorrAvg[i]);
382  outputFile << std::endl;
383  }
384  outputFile.close();
385 
386  // Sum over autocorrelation (only positive time lags)
387  // Count first element with a factor 1/2, due to symmetry
388  double acSum = 0;
389  acSum = 0.5*autoCorrAvg[0];
390  for (int i = 1; i < autoCorrCapacity_; i++) {
391  acSum += autoCorrAvg[i];
392  }
393 
394  // Write sum in .ave file
395  fileName = outputFileName_;
396  fileName += ".ave";
397  system().fileMaster().openOutputFile(fileName, outputFile);
398  outputFile << "autoCorrSum " << acSum << std::endl;
399  outputFile.close();
400  }
401  }
