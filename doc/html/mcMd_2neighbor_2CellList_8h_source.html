364  /*
365  * Return index of the cell that contains the position array pos = {x, y, z}.
366  */
367  inline int CellList::cellIndexFromPosition(const Vector &pos) const
368  {
369  int cx, cy, cz;
370 
371  {
372  Vector posG;
373  boundaryPtr_->transformCartToGen(pos, posG);
374  cx = int(posG[0]*invCellWidths_[0]);
375  cy = int(posG[1]*invCellWidths_[1]);
376  cz = int(posG[2]*invCellWidths_[2]);
377  }
378 
379  assert(cx >= 0);
380  assert(cx < numCells_[0]);
381  assert(cy >= 0);
382  assert(cy < numCells_[1]);
383  assert(cz >= 0);
384  assert(cz < numCells_[2]);
385 
386  return cz + cy*numCells_[2] + cx*YZCells_;
387  }
