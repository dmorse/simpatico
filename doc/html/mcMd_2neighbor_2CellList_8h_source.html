357  /*
358  * Return index of the cell that contains the position array pos = {x, y, z}.
359  */
360  inline int CellList::cellIndexFromPosition(const Vector &pos) const
361  {
362  int cx, cy, cz;
363 
364  {
365  Vector posG;
366  boundaryPtr_->transformCartToGen(pos, posG);
367  cx = int(posG[0]*invCellWidths_[0]);
368  cy = int(posG[1]*invCellWidths_[1]);
369  cz = int(posG[2]*invCellWidths_[2]);
370  }
371 
372  assert(cx >= 0);
373  assert(cx < numCells_[0]);
374  assert(cy >= 0);
375  assert(cy < numCells_[1]);
376  assert(cz >= 0);
377  assert(cz < numCells_[2]);
378 
379  return cz + cy*numCells_[2] + cx*YZCells_;
380  }
