366  /*
367  * Return value of offset for cell a distance x along axis i shifted to the range
368  * 0 <= x <= grid_.Dimension(i)
369  */
370  inline int CellList::calculateAxisOffset(int cellId, int i, int x) const
371  {
372  // mod the shift so it's within the grid dimension
373  x %= grid_.dimension(i);
374 
375  // calculate cell's position within the grid
376  int p[3]; int j;
377  for (j = 0; j < Dimension -1; ++j) {
378  p[j] = cellId/gridOffsets_[j];
379  cellId -= p[j]*gridOffsets_[j];
380  }
381  p[j] = cellId;
382 
383  if (p[i] + x >= grid_.dimension(i)) return (x - grid_.dimension(i))*gridOffsets_[i];
384  if (p[i] + x < 0) return (grid_.dimension(i) + x)*gridOffsets_[i];
385  return x*gridOffsets_[i];
386  }

390  /*
391  * Identify the cell within which a position lies.
392  */
393  inline int CellList::cellIndexFromPosition(const Vector& position) const
394  {
395  IntVector r;
396  for (int i = 0; i < Dimension; ++i) {
397  if (position[i] <= lower_[i]) {
398  return -1;
399  }
400  if (position[i] >= upper_[i]) {
401  return -1;
402  }
403  r[i] = int( (position[i] - lower_[i]) / cellLengths_[i] );
404  assert(r[i] < grid_.dimension(i));
405  }
406  return grid_.rank(r);
407  }
