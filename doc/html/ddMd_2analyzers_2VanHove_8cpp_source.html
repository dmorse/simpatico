147  void VanHove::sample(long iStep)
148  {
149  if (isAtInterval(iStep)) {
150 
151  Vector position;
152  std::complex<double> expFactor;
153  double product, coeff;
154  AtomIterator atomIter;
155  int i, typeId;
156 
157  makeWaveVectors();
158 
159  // Set all Fourier modes to zero
160  for (i = 0; i < nWave_; ++i) {
161  fourierModes_[i] = std::complex<double>(0.0, 0.0);
162  }
163 
164  simulation().atomStorage().begin(atomIter);
165  for ( ; atomIter.notEnd(); ++atomIter) {
166  position = atomIter->position();
167  typeId = atomIter->typeId();
168  coeff = atomTypeCoeffs_[typeId];
169 
170  // Loop over wavevectors
171  for (i = 0; i < nWave_; ++i) {
172 
173  product = position.dot(waveVectors_[i]);
174  expFactor = exp( product*Constants::Im );
175  fourierModes_[i] += coeff*expFactor;
176  }
177  }
178 
179  for (i = 0; i < nWave_; ++i) {
180  totalFourierModes_[i] = std::complex<double>(0.0, 0.0);
181  }
182 
183  #ifdef UTIL_MPI
184  // Loop over wavevectors
185  for (int i = 0; i < nWave_; ++i) {
186  //Sum values from all processors.
187  simulation().domain().communicator().
188  Reduce(&fourierModes_[i], &totalFourierModes_[i],
189  1, MPI::DOUBLE_COMPLEX, MPI::SUM, 0);
190  }
191  #else
192  for (int i = 0; i < nWave_; ++i) {
193  totalFourierModes_[i] = fourierModes_[i];
194  }
195  #endif
196 
197  if (simulation().domain().isMaster()) {
198  // Add Fourier modes to autocorrelation accumulators
199  double sqrtV = sqrt(simulation().boundary().volume());
200  for (int i = 0; i < nWave_; ++i) {
201  accumulators_[i].sample(totalFourierModes_[i]/sqrtV);
202  }
203  }
204  ++nSample_;
205  }

213  {

216  int i, j;
217 
218  // Calculate wavevectors
