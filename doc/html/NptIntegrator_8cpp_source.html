144  /*
145  * First half of velocity-verlet update.
146  */
147  void NptIntegrator::integrateStep1()
148  {
149  Vector dv;
150  double prefactor; // = 0.5*dt/mass
151  double xi_prime;
152  AtomIterator atomIter;
153  Simulation& sim = simulation();
154 
155  sim.computeVirialStress();
156  sim.computeKineticStress();
157  sim.computeKineticEnergy();
158 
159  if (sim.domain().isMaster()) {
160  T_target_ = sim.energyEnsemble().temperature();
161  P_target_ = sim.boundaryEnsemble().pressure();
162  T_kinetic_ = sim.kineticEnergy()*2.0/ndof_;
163  Tensor stress = sim.virialStress();
164  stress += sim.kineticStress();
165 
166  P_curr_diag_ = Vector(stress(0, 0), stress(1,1), stress(2,2));
167  double P_curr = (1.0/3.0)*stress.trace();
168 
169  double W = (1.0/2.0)*ndof_*T_target_*tauP_*tauP_;
170  double mtk_term = (1.0/2.0)*dt_*T_kinetic_/W;
171 
172  // Advance barostat (first half of update)
173  double V = sim.boundary().volume();
174  if (mode_ == Cubic) {
175  nu_[0] += (1.0/2.0)*dt_*V/W*(P_curr - P_target_) + mtk_term;
176  nu_[1] = nu_[2] = nu_[0];
177  } else if (mode_ == Tetragonal) {
178  nu_[0] += (1.0/2.0)*dt_*V/W*(P_curr_diag_[0] - P_target_) + mtk_term;
179  nu_[1] += (1.0/2.0)*dt_*V/W*((1.0/2.0)*(P_curr_diag_[1] + P_curr_diag_[2]) - P_target_)
180  + mtk_term;
181  nu_[2] = nu_[1];
182  } else if (mode_ == Orthorhombic) {
183  nu_[0] += (1.0/2.0)*dt_*V/W*(P_curr_diag_[0] - P_target_) + mtk_term;
184  nu_[1] += (1.0/2.0)*dt_*V/W*(P_curr_diag_[1] - P_target_) + mtk_term;
185  nu_[2] += (1.0/2.0)*dt_*V/W*(P_curr_diag_[2] - P_target_) + mtk_term;
186  }
187 
188  // Advance thermostat
189  xi_prime = xi_ + (1.0/4.0)*dt_/tauT_/tauT_*(T_kinetic_/T_target_ - 1.0);
190  xi_ = xi_prime + (1.0/4.0)*dt_/tauT_/tauT_*(T_kinetic_/T_target_*exp(-xi_prime*dt_) - 1.0);
191  eta_ += (1.0/2.0)*xi_prime*dt_;
192  }
193 
194  #ifdef UTIL_MPI
195  bcast(domain().communicator(), xi_, 0);
196  bcast(domain().communicator(), xi_prime, 0);
197  bcast(domain().communicator(), nu_, 0);
198  #endif
199 
200  // Precompute loop invariant quantities
201  mtk_term_2_ = (nu_[0]+nu_[1]+nu_[2])/ndof_;
202  Vector v_fac = Vector((1.0/4.0)*(nu_[0]+mtk_term_2_),
203  (1.0/4.0)*(nu_[1]+mtk_term_2_),
204  (1.0/4.0)*(nu_[2]+mtk_term_2_));
205  exp_v_fac_ = Vector(exp(-v_fac[0]*dt_),
206  exp(-v_fac[1]*dt_),
207  exp(-v_fac[2]*dt_));
208  Vector exp_v_fac_2 = Vector(exp(-(2.0*v_fac[0]+(1.0/2.0)*xi_prime)*dt_),
209  exp(-(2.0*v_fac[1]+(1.0/2.0)*xi_prime)*dt_),
210  exp(-(2.0*v_fac[2]+(1.0/2.0)*xi_prime)*dt_));
211  Vector r_fac = Vector((1.0/2.0)*nu_[0],
212  (1.0/2.0)*nu_[1],
213  (1.0/2.0)*nu_[2]);
214  Vector exp_r_fac = Vector(exp(r_fac[0]*dt_),
215  exp(r_fac[1]*dt_),
216  exp(r_fac[2]*dt_));
217 
218  //Coefficients of sinh(x)/x = a_0 + a_2 * x^2 + a_4 * x^4 + a_6 * x^6 + a_8 * x^8 + a_10 * x^10
219  const double a[] = { double(1.0), double(1.0/6.0), double(1.0/120.0),
220  double(1.0/5040.0), double(1.0/362880.0), double(1.0/39916800.0)};
221 
222  Vector arg_v = Vector(v_fac[0]*dt_, v_fac[1]*dt_, v_fac[2]*dt_);
223  Vector arg_r = Vector(r_fac[0]*dt_, r_fac[1]*dt_, r_fac[2]*dt_);
224 
225  sinhx_fac_v_ = Vector(0.0, 0.0, 0.0);
226  Vector sinhx_fac_r = Vector(0.0, 0.0, 0.0);
227 
228  Vector term_v = Vector(1.0, 1.0, 1.0);
229  Vector term_r = Vector(1.0, 1.0, 1.0);
230 
231  for (unsigned int i = 0; i < 6; i++) {
232  sinhx_fac_v_ += Vector(a[0]*term_v[0],
233  a[1]*term_v[1],
234  a[2]*term_v[2]);
235  sinhx_fac_r += Vector(a[0]*term_r[0],
236  a[1]*term_r[1],
237  a[2]*term_r[2]);
238  term_v = Vector(term_v[0] * arg_v[0] * arg_v[0],
239  term_v[1] * arg_v[1] * arg_v[1],
240  term_v[2] * arg_v[2] * arg_v[2]);
241  term_r = Vector(term_r[0] * arg_r[0] * arg_r[0],
242  term_r[1] * arg_r[1] * arg_r[1],
243  term_r[2] * arg_r[2] * arg_r[2]);
244  }
245 
246  // 1st half of NPT
247  Vector vtmp;
248  atomStorage().begin(atomIter);
249  for ( ; atomIter.notEnd(); ++atomIter) {
250  prefactor = prefactors_[atomIter->typeId()];
251 
252  dv.multiply(atomIter->force(), prefactor);
253  dv[0] = dv[0] * exp_v_fac_[0]*sinhx_fac_v_[0];
254  dv[1] = dv[1] * exp_v_fac_[1]*sinhx_fac_v_[1];
255  dv[2] = dv[2] * exp_v_fac_[2]*sinhx_fac_v_[2];
256 
257  Vector& v = atomIter->velocity();
258  v[0] = v[0] * exp_v_fac_2[0] + dv[0];
259  v[1] = v[1] * exp_v_fac_2[1] + dv[1];
260  v[2] = v[2] * exp_v_fac_2[2] + dv[2];
261 
262  vtmp[0] = v[0]*exp_r_fac[0] *sinhx_fac_r[0];
263  vtmp[1] = v[1]*exp_r_fac[1] *sinhx_fac_r[1];
264  vtmp[2] = v[2]*exp_r_fac[2] *sinhx_fac_r[2];
265 
266  Vector& r = atomIter->position();
267  r[0] = r[0]*exp_r_fac[0]*exp_r_fac[0] + vtmp[0]*dt_;
268  r[1] = r[1]*exp_r_fac[1]*exp_r_fac[1] + vtmp[1]*dt_;
269  r[2] = r[2]*exp_r_fac[2]*exp_r_fac[2] + vtmp[2]*dt_;
270  }
271 
272  // Advance box lengths
273  Vector box_len_scale = Vector(exp(nu_[0]*dt_),
274  exp(nu_[1]*dt_),
275  exp(nu_[2]*dt_));
276 
277  Vector L = sim.boundary().lengths();
278  L[0] *= box_len_scale[0];
279  L[1] *= box_len_scale[1];
280  L[2] *= box_len_scale[2];
281 
282  // Update box lengths
283  sim.boundary().setOrthorhombic(L);
284  }

286  void NptIntegrator::integrateStep2()
287  {
288  Vector dv;
289  double prefactor; // = 0.5*dt/mass
290  AtomIterator atomIter;
291  Simulation& sim = simulation();
292 
293  Vector v_fac_2 = Vector((1.0/2.0)*(nu_[0]+mtk_term_2_),
294  (1.0/2.0)*(nu_[1]+mtk_term_2_),
295  (1.0/2.0)*(nu_[2]+mtk_term_2_));
296  Vector exp_v_fac_2 = Vector(exp(-v_fac_2[0]*dt_),
297  exp(-v_fac_2[1]*dt_),
298  exp(-v_fac_2[2]*dt_));
299 
300  double v2_sum = 0.0;
301 
302  // 2nd half of NPT
303  atomStorage().begin(atomIter);
304  for ( ; atomIter.notEnd(); ++atomIter) {
305  prefactor = prefactors_[atomIter->typeId()];
306  dv.multiply(atomIter->force(), prefactor);
307  dv[0] = dv[0] * exp_v_fac_[0]*sinhx_fac_v_[0];
308  dv[1] = dv[1] * exp_v_fac_[1]*sinhx_fac_v_[1];
309  dv[2] = dv[2] * exp_v_fac_[2]*sinhx_fac_v_[2];
310 
311  Vector& v = atomIter->velocity();
312  v[0] = v[0] * exp_v_fac_2[0] + dv[0];
313  v[1] = v[1] * exp_v_fac_2[1] + dv[1];
314  v[2] = v[2] * exp_v_fac_2[2] + dv[2];
315 
316  // total up 2*kinetic energy
317  double mass = simulation().atomType(atomIter->typeId()).mass();
318  v2_sum += v.dot(v)*mass;
319  }
320 
321  // Advance thermostat (second half of update)
322  double T_prime;
323  double xi_prime;
324  #ifdef UTIL_MPI
325  //reduce vs_sum
326  domain().communicator().Reduce(&v2_sum, &T_prime, 1, MPI::DOUBLE, MPI::SUM,0);
327  #endif
328 
329  if (domain().isMaster()) {
330  T_prime /= ndof_;
331  xi_prime = xi_ + (1.0/4.0)*dt_/tauT_/tauT_*(T_prime/T_target_ - 1.0);
332  xi_ = xi_prime + (1.0/4.0)*dt_/tauT_/tauT_*(T_prime/T_target_*exp(-xi_prime*dt_) - 1.0);
333  eta_ += (1.0/2.0)*xi_prime*dt_;
334  }
335 
336  #ifdef UTIL_MPI
337  bcast(domain().communicator(), xi_prime,0);
338  #endif
339 
340  // Rescale velocities
341  double exp_v_fac_thermo = exp(-(1.0/2.0)*xi_prime*dt_);
342  atomStorage().begin(atomIter);
343  for ( ; atomIter.notEnd(); ++atomIter) {
344  atomIter->velocity().multiply(atomIter->velocity(), exp_v_fac_thermo);
345  }
346 
347  sim.velocitySignal().notify();
348  sim.computeKineticStress();
349  sim.computeKineticEnergy();
350  sim.computeVirialStress();
351 
352  // Advance barostat
353  if (sim.domain().isMaster()) {
354  T_kinetic_ = sim.kineticEnergy()*2.0/ndof_;
355  Tensor stress = sim.virialStress();
356  stress += sim.kineticStress();
357 
358  P_curr_diag_ = Vector(stress(0,0), stress(1,1), stress(2,2));
359  double P_curr = (1.0/3.0)*stress.trace();
360 
361  double W = (1.0/2.0)*ndof_*T_target_*tauP_*tauP_;
362  double mtk_term = (1.0/2.0)*dt_*T_kinetic_/W;
363 
364  double V = sim.boundary().volume();
365  if (mode_ == Cubic) {
366  nu_[0] += (1.0/2.0)*dt_*V/W*(P_curr - P_target_) + mtk_term;
367  nu_[1] = nu_[2] = nu_[0];
368  } else if (mode_ == Tetragonal) {
369  nu_[0] += (1.0/2.0)*dt_*V/W*(P_curr_diag_[0] - P_target_) + mtk_term;
370  nu_[1] += (1.0/2.0)*dt_*V/W*((1.0/2.0)*(P_curr_diag_[1]+P_curr_diag_[2]) - P_target_) + mtk_term;
371  nu_[2] = nu_[1];
372  } else if (mode_ == Orthorhombic) {
373  nu_[0] += (1.0/2.0)*dt_*V/W*(P_curr_diag_[0] - P_target_) + mtk_term;
374  nu_[1] += (1.0/2.0)*dt_*V/W*(P_curr_diag_[1] - P_target_) + mtk_term;
375  nu_[2] += (1.0/2.0)*dt_*V/W*(P_curr_diag_[2] - P_target_) + mtk_term;
376  }
377  }
