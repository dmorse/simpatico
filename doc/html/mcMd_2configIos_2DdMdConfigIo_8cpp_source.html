43  void DdMdConfigIo::read(std::istream &in)
44  {
45  // Calculate atom and bond capacities for entire simulation
46  int atomCapacity = 0;
47  int bondCapacity = 0;
48  int nSpecies = simulation().nSpecies();
49  int speciesCapacity = 0;
50  int iSpec;
51  Species* speciesPtr;
52  for (iSpec = 0; iSpec < nSpecies; ++iSpec) {
53  speciesPtr = &simulation().species(iSpec);
54  speciesCapacity = speciesPtr->capacity();
55  atomCapacity += speciesCapacity*speciesPtr->nAtom();
56  #ifdef INTER_BOND
57  bondCapacity += speciesCapacity*speciesPtr->nBond();
58  #endif
59  }
60 
61  // Read boundary
62  in >> Label("BOUNDARY");
63  in >> system().boundary();
64 
65  // Read atomic positions
66  // Atom tags must appear in order, numbered from 0
67  // Number of atoms must match atomCapacity.
68  in >> Label("ATOMS");
69  int nAtom;
70  in >> Label("nAtom") >> nAtom;
71  if (nAtom != atomCapacity) {
72  UTIL_THROW("nAtom != atomCapacity");
73  }
74  Molecule* molPtr;
75  Molecule::AtomIterator atomIter;
76  int iMol, iAtom, nMolecule, atomId, atomTypeId;
77  int sId, mId, aId;
78  for (iSpec=0; iSpec < nSpecies; ++iSpec) {
79  speciesPtr = &simulation().species(iSpec);
80  nMolecule = speciesPtr->capacity();
81  for (iMol = 0; iMol < nMolecule; ++iMol) {
82  molPtr = &(simulation().getMolecule(iSpec));
83  system().addMolecule(*molPtr);
84 
85  // Read positions
86  iAtom = 0;
87  for (molPtr->begin(atomIter); atomIter.notEnd(); ++atomIter) {
88 
89  in >> atomId >> atomTypeId;
90  if (atomId != atomIter->id()) {
91  UTIL_THROW("Atom tags not ordered");
92  }
93  if (hasMolecules_) {
94  in >> sId >> mId >> aId;
95  if (sId != iSpec) {
96  UTIL_THROW("Invalid species id");
97  }
98  if (mId != iMol) {
99  UTIL_THROW("Invalid molecule id");
100  }
101  if (aId != iAtom) {
102  UTIL_THROW("Invalid local atom id");
103  }
104  }
105 
106  in >> atomIter->position();
107  in >> atomIter->velocity();
108 
109  ++iAtom;
110  }
111  }
112  }

116  void DdMdConfigIo::write(std::ostream &out)
117  {
118  // Count total numbers of atoms and bonds in all species.
119  Species *speciesPtr;
120  int iSpec, nMolecule;
121  int nAtom = 0;
122  #ifdef INTER_BOND
123  int nBond = 0;
124  #endif
125  #ifdef INTER_ANGLE
126  int nAngle = 0;
127  #endif
128  #ifdef INTER_DIHEDRAL
129  int nDihedral = 0;
130  #endif
131  for (iSpec = 0; iSpec < simulation().nSpecies(); ++iSpec) {
132  speciesPtr = &simulation().species(iSpec);
133  nMolecule = system().nMolecule(iSpec);
134  nAtom += nMolecule*(speciesPtr->nAtom());
135  #ifdef INTER_BOND
136  nBond += nMolecule*(speciesPtr->nBond());
137  #endif
138  #ifdef INTER_ANGLE
139  nAngle += nMolecule*(speciesPtr->nAngle());
140  #endif
141  #ifdef INTER_DIHEDRAL
142  nDihedral += nMolecule*(speciesPtr->nDihedral());
143  #endif
144  }
145 
146  // Write boundary
147  out << "BOUNDARY" << std::endl;
148  out << system().boundary() << std::endl;
149  out << std::endl;
150 
151  // Write atoms
152  out << "ATOMS" << std::endl;
153  out << "nAtom " << nAtom << std::endl;
154  System::MoleculeIterator molIter;
155  Molecule::AtomIterator atomIter;
156  int i = 0;
157  int iMol, iAtom;
158  for (iSpec=0; iSpec < simulation().nSpecies(); ++iSpec) {
159  iMol = 0;
160  system().begin(iSpec, molIter);
161  for ( ; molIter.notEnd(); ++molIter) {
162  iAtom = 0;
163  molIter->begin(atomIter);
164  for ( ; atomIter.notEnd(); ++atomIter) {
165  out << Int(atomIter->id(), 10);
166  out << Int(atomIter->typeId(), 5);
167  if (hasMolecules_) {
168  out << Int(iSpec, 6);
169  out << Int(iMol, 10);
170  out << Int(iAtom, 6);
171  }
172  out << "\n" << atomIter->position();
173  out << "\n" << atomIter->velocity();
174  out << "\n";
175  ++i;
176  ++iAtom;
177  }
178  ++iMol;
179  }
180  }
181  out << std::endl;
182 
183  #ifdef INTER_BOND
184  // Write Bonds
185  out << "BONDS" << std::endl;
186  out << "nBond " << nBond << std::endl;
187  Molecule::BondIterator bondIter;
188  i = 0;
189  for (iSpec=0; iSpec < simulation().nSpecies(); ++iSpec) {
190  if (system().simulation().species(iSpec).nBond() > 0) {
191  system().begin(iSpec, molIter);
192  for ( ; molIter.notEnd(); ++molIter) {
193  molIter->begin(bondIter);
194  for ( ; bondIter.notEnd(); ++bondIter) {
195  out << Int(i, 8) << Int(bondIter->typeId(), 5);
196  out << Int(bondIter->atom(0).id(), 10);
197  out << Int(bondIter->atom(1).id(), 10);
198  out << std::endl;
199  ++i;
200  }
201  }
202  }
203  }
204  out << std::endl;
205  #endif
206 
207  #ifdef INTER_ANGLE
208  // Write Angles
209  out << "ANGLES" << std::endl;
210  out << "nAngle " << nAngle << std::endl;
211  Molecule::AngleIterator angleIter;
212  i = 0;
