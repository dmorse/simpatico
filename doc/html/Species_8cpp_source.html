472  /*
473  * Check validity of Species. Return true if valid, or throw an Exception.
474  */
475  bool Species::isValid() const
476  {
477 
478  if (reservoir_.isAllocated()) {
479 
480  // Check validity of reservoir (throws exception if invalid)
481  reservoir_.isValid();
482 
483  // Check atomTypeIds array
484  if (!atomTypeIds_.isAllocated()) {
485  UTIL_THROW("AtomTypeIds_ not allocated");
486  }
487  if (!isMutable()) {
488  for (int i = 0; i < nAtom_; ++i) {
489  if (atomTypeIds_[i] < 0) {
490  UTIL_THROW("Negative AtomTypeIds_ in non-mutable Species");
491  }
492  }
493  }
494 
495  // Loop over all bonds (if any) in speciesBonds_ array
496  int atomId, bondId, atomId0, atomId1, j;
497  bool hasBond;
498  for (bondId = 0; bondId < nBond_; ++bondId) {
499 
500  atomId0 = speciesBond(bondId).atomId(0);
501  atomId1 = speciesBond(bondId).atomId(1);
502 
503  // Check that atomIds are in valid range and unequal
504  if (atomId0 < 0 || atomId0 >= nAtom_) {
505  UTIL_THROW("Invalid atomId0 in speciesBonds_");
506  }
507  if (atomId1 < 0 || atomId1 >= nAtom_) {
508  UTIL_THROW("Invalid atomId1 in speciesBonds_");
509  }
510  if (atomId0 == atomId1) {
511  UTIL_THROW("Equal atom ids in a SpeciesBond");
512  }
513 
514  // Check that bondId is in atombondIdArrays_ for atomId0
515  hasBond = false;
516  for (j = 0; j < atomBondIdArrays_[atomId0].size(); ++j) {
517  if (atomBondIdArrays_[atomId0][j] == bondId) {
518  hasBond = true;
519  break;
520  }
521  }
522  if (!hasBond) {
523  UTIL_THROW("BondId missing from atomBondIdArrays_");
524  }
525 
526  // Check that bondId is in atombondIdArrays_ for atomId0
527  hasBond = false;
528  for (j = 0; j < atomBondIdArrays_[atomId1].size(); ++j) {
529  if (atomBondIdArrays_[atomId1][j] == bondId) {
530  hasBond = true;
531  break;
532  }
533  }
534  if (!hasBond) {
535  UTIL_THROW("BondId missing from atomBondIdArrays_");
536  }
537 
538  }
539 
540  // Loop over atomBondIdArrays for all atoms
541  int bondCounter = 0;
542  for (atomId = 0; atomId < nAtom_; ++atomId) {
543  for (j = 0; j < atomBondIdArrays_[atomId].size(); ++j) {
544  bondId = atomBondIdArrays_[atomId][j];
545 
546  if (bondId < 0 || bondId >= nBond_)
547  UTIL_THROW("Bond index out of range");
548 
549  atomId0 = speciesBond(bondId).atomId(0);
550  atomId1 = speciesBond(bondId).atomId(1);
551  if (atomId != atomId0 && atomId != atomId1)
552  UTIL_THROW("Inconsistency in atomBondId and bond");
553  ++bondCounter;
554 
555  }
556  }
557  if (bondCounter != 2*nBond_)
558  UTIL_THROW("Inconsistency in total number of bonds");
559 
560  #ifdef INTER_ANGLE
561  // Loop over all angles (if any) in speciesAngles_ array
562  int angleId, id, id2, atomId2(-1);
563  bool hasAngle;
564  for (angleId = 0; angleId < nAngle_; ++angleId) {
565 
566  for (id = 0; id < 3; ++id) {
567  atomId = speciesAngle(angleId).atomId(id);
568 
569  // Check that atomIds are in valid range and unequal
570  if (atomId < 0 || atomId >= nAtom_) {
571  UTIL_THROW("Invalid atomId in speciesAngles_");
572  }
573  for (id2 = id + 1; id2 < 3; ++id2) {
574  atomId2 = speciesAngle(angleId).atomId(id2);
575  if (atomId2 == atomId) {
576  UTIL_THROW("Equal atom ids in a SpeciesAngle");
577  }
578  }
579 
580  // Check that angleId is in atomAngleIdArrays_ for atomId
581  hasAngle = false;
582  for (j = 0; j < atomAngleIdArrays_[atomId].size(); ++j) {
583  if (atomAngleIdArrays_[atomId][j] == angleId) {
584  hasAngle = true;
585  break;
586  }
587  }
588  if (!hasAngle) {
589  UTIL_THROW("AngleId missing from atomAngleIdArrays_");
590  }
591  }
592 
593  }
594 
595  // Loop over atomAngleIdArrays for all atoms
596  int angleCounter = 0;
597  bool hasAtom;
598  for (atomId = 0; atomId < nAtom_; ++atomId) {
599  for (j = 0; j < atomAngleIdArrays_[atomId].size(); ++j) {
600  angleId = atomAngleIdArrays_[atomId][j];
601  if (angleId < 0 || angleId >= nAngle_)
602  UTIL_THROW("Angle index out of range");
603 
604  hasAtom = false;
605  for (id = 0; id < 3; ++id) {
606  atomId2 = speciesAngle(angleId).atomId(id);
607  if (atomId2 == atomId) {
608  hasAtom = true;
609  break;
610  }
611  }
612  if (!hasAtom) {
613  UTIL_THROW("Inconsistency in atomAngleId and angle");
614  }
615  ++angleCounter;
616  }
617  }
618  if (angleCounter != 3*nAngle_)
619  UTIL_THROW("Inconsistency in total number of angles");
620  #endif
621 
622  #ifdef INTER_DIHEDRAL
623  // Loop over all dihedrals (if any) in speciesDihedrals_ array
624  int dihedralId, tId, tId2, tAtomId, tAtomId2;
625  bool hasDihedral;
626 
627  for (dihedralId = 0; dihedralId < nDihedral_; ++dihedralId) {
628 
629  for (tId = 0; tId < 4; ++tId) {
630  tAtomId = speciesDihedral(dihedralId).atomId(tId);
631 
632  // Check that atomIds are in valid range and unequal
633  if (tAtomId < 0 || tAtomId >= nAtom_) {
634  UTIL_THROW("Invalid atomId in speciesDihedral_");
635  }
636  for (tId2 = tId + 1; tId2 < 4; ++tId2) {
637  tAtomId2 = speciesDihedral(dihedralId).atomId(tId2);
638  if (tAtomId2 == tAtomId) {
639  UTIL_THROW("Equal atom ids in a SpeciesDihedral");
640  }
641  }
642 
643  // Check that dihedralId is in atomDihedralIdArrays_ for tAtomId
644  hasDihedral = false;
645  for (j = 0; j < atomDihedralIdArrays_[tAtomId].size(); ++j) {
646  if (atomDihedralIdArrays_[tAtomId][j] == dihedralId) {
647  hasDihedral = true;
648  break;
649  }
650  }
651  if (!hasDihedral) {
652  UTIL_THROW("DihedralId missing from atomDihedralIdArrays_");
653  }
654  }
655 
656  }
657 
658  // Loop over atomDihedralIdArrays for all atoms.
659  int dihedralCounter = 0;
660  bool tHasAtom;
661  for (tAtomId = 0; tAtomId < nAtom_; ++tAtomId) {
662  for (j = 0; j < atomDihedralIdArrays_[tAtomId].size(); ++j) {
663  dihedralId = atomDihedralIdArrays_[tAtomId][j];
664  if (dihedralId < 0 || dihedralId >= nDihedral_)
665  UTIL_THROW("Dihedral index out of range");
666
