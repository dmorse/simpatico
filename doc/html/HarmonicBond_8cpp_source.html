108  /*
109  * Generate a random bond length chosen from an equilibrium distribution for
110  * randomly oriented bonds.
111  *
112  * The desired probability distribution is proportional to x*x*G(x) for x > 0,
113  * where G(x) is a Gaussian distribution with a mean x0=length_[type] and a
114  * standard deviation sd = sqrt(temperature/kappa_[type]).
115  *
116  * Algorithm:
117  * - Generate trial bond length x from the Gaussian distribution G(x).
118  * - If x < 0, generate another trial bond length.
119  * - Let y = x*x/xm*xm, where xm = x0 + 7.0*sd is a maximum length.
120  * The algorithm is correct only for y < 1, or x < xm. The distribution
121  * G(y) for generated values of y is proportional to G(x)/x.
122  * - Accept y if it is greater than a uniform random number in [0,1].
123  * For y < 1, or x < xm, the probability of accepting y is equal to y.
124  * The distribution A(y) of accepted values of y for y < 1 is thus
125  * proportional to y*G(y), or to x*G(x).
126  * - If y is accepted, return x. The distribution A(x) for accepted
127  * values of x is proportional to x*x*G(x) for x < xm.
128  *
129  * Limitations:
130  * The distribution is correct only for x < xm. The probability of
131  * generating a value x >= xm is, however, less than exp(-49/2).
132  */
133  double HarmonicBond::randomBondLength(
134  Random *random, double beta, int type) const
135  {
136 
137  double sd = 1.0/sqrt( beta*kappa_[type] );
138  double x0 = length_[type];
139  double xm2 = length_[type] + 7.0*sd;
140  xm2 = xm2*xm2;
141  double x;
142 
143  // Generate trials until one is accepted
144  for (;;) {
145  x = x0 + sd*random->gaussian();
146  if (x < 0) continue;
147  if ( x*x/xm2 >= random->uniform(0.0,1.0) ) {
148  return x;
149  };
150  }
151  }
