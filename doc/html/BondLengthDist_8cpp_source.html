101  void BondLengthDist::sample(long iStep)
102  {

105  System::MoleculeIterator molIter;
106  Molecule::BondIterator bondIter;
107  double lsq, l;

110  for (molIter->begin(bondIter); bondIter.notEnd(); ++bondIter) {
111  lsq = system().boundary().distanceSq( bondIter->atom(0).position(),
112  bondIter->atom(1).position());
